Passage Planning sayfasını tamamen yeniden tasarla. Tam otomatik, broker'ların kolayca kullanabileceği bir sistem olacak. Elle pilotlama yok — gemi seç, limanları seç, hız belirle, gerisini sistem hesaplasın.

## ANA AKIŞ

1. Gemi seç (IMO ile ara → Datalastic'ten otomatik bilgi doldur)
2. Kalkış limanını seç (port search)
3. Varış limanını seç (port search)
4. Ara limanlar ekle (opsiyonel)
5. Hız belirle (knot)
6. SİSTEM OTOMATİK HESAPLASIN:
   - Toplam mesafe (NM)
   - Tahmini süre (gün/saat)
   - Tahmini yakıt tüketimi (HFO/MGO ton)
   - Geçilecek boğazlar/kanallar
   - Waypoint'ler (otomatik rota noktaları)
   - Her leg için mesafe, süre, yakıt
   - ETA her liman için

// POST /calculate-route — İki liman arası otomatik rota hesapla
router.post("/calculate-route", isAuthenticated, async (req: any, res) => {
  try {
    const { departurePort, destinationPort, intermediatePortss, plannedSpeed, departureLat, departureLon, destinationLat, destinationLon } = req.body;
    
    // Boğaz ve kanal veritabanı
    const STRAITS_AND_CANALS = [
      { name: "Bosphorus (İstanbul Boğazı)", lat: 41.12, lon: 29.05, radius: 20, transitTimeHours: 2, type: "strait" },
      { name: "Dardanelles (Çanakkale Boğazı)", lat: 40.20, lon: 26.40, radius: 30, transitTimeHours: 3, type: "strait" },
      { name: "Suez Canal", lat: 30.45, lon: 32.35, radius: 50, transitTimeHours: 14, type: "canal" },
      { name: "Panama Canal", lat: 9.08, lon: -79.68, radius: 30, transitTimeHours: 10, type: "canal" },
      { name: "Strait of Gibraltar", lat: 35.97, lon: -5.50, radius: 20, transitTimeHours: 1.5, type: "strait" },
      { name: "Strait of Hormuz", lat: 26.56, lon: 56.25, radius: 30, transitTimeHours: 2, type: "strait" },
      { name: "Strait of Malacca", lat: 2.50, lon: 101.50, radius: 100, transitTimeHours: 12, type: "strait" },
      { name: "Strait of Messina", lat: 38.20, lon: 15.63, radius: 10, transitTimeHours: 1, type: "strait" },
      { name: "English Channel", lat: 50.50, lon: 0.50, radius: 50, transitTimeHours: 6, type: "strait" },
      { name: "Kiel Canal", lat: 54.15, lon: 9.70, radius: 30, transitTimeHours: 8, type: "canal" },
      { name: "Cape of Good Hope", lat: -34.35, lon: 18.50, radius: 50, transitTimeHours: 0, type: "waypoint" },
      { name: "Strait of Bonifacio", lat: 41.25, lon: 9.15, radius: 15, transitTimeHours: 1, type: "strait" },
      { name: "Strait of Sicily", lat: 37.00, lon: 11.50, radius: 40, transitTimeHours: 3, type: "strait" },
      { name: "Bab el-Mandeb", lat: 12.58, lon: 43.33, radius: 20, transitTimeHours: 2, type: "strait" },
      { name: "Strait of Taiwan", lat: 24.00, lon: 119.50, radius: 50, transitTimeHours: 4, type: "strait" },
      { name: "Torres Strait", lat: -10.50, lon: 142.00, radius: 30, transitTimeHours: 3, type: "strait" },
      { name: "Danish Straits (Øresund)", lat: 55.90, lon: 12.70, radius: 20, transitTimeHours: 2, type: "strait" },
    ];
    
    // ECA (Emission Control Area) bölgeleri
    const ECA_ZONES = [
      { name: "Baltic Sea ECA", latMin: 53, latMax: 66, lonMin: 10, lonMax: 30 },
      { name: "North Sea ECA", latMin: 48, latMax: 62, lonMin: -5, lonMax: 10 },
      { name: "North America ECA", latMin: 25, latMax: 50, lonMin: -130, lonMax: -60 },
      { name: "US Caribbean ECA", latMin: 17, latMax: 32, lonMin: -98, lonMax: -64 },
      { name: "Mediterranean Sea (2025 SECA)", latMin: 30, latMax: 46, lonMin: -6, lonMax: 36 },
    ];
    
    // Haversine mesafe hesaplama (NM)
    function haversineNm(lat1: number, lon1: number, lat2: number, lon2: number): number {
      const R = 3440.065; // Dünya yarıçapı nautical mil
      const toRad = (d: number) => d * Math.PI / 180;
      const dLat = toRad(lat2 - lat1);
      const dLon = toRad(lon2 - lon1);
      const a = Math.sin(dLat/2)**2 + Math.cos(toRad(lat1)) * Math.cos(toRad(lat2)) * Math.sin(dLon/2)**2;
      return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));
    }
    
    // İki nokta arası bearing (derece)
    function bearing(lat1: number, lon1: number, lat2: number, lon2: number): number {
      const toRad = (d: number) => d * Math.PI / 180;
      const toDeg = (r: number) => r * 180 / Math.PI;
      const dLon = toRad(lon2 - lon1);
      const y = Math.sin(dLon) * Math.cos(toRad(lat2));
      const x = Math.cos(toRad(lat1)) * Math.sin(toRad(lat2)) - Math.sin(toRad(lat1)) * Math.cos(toRad(lat2)) * Math.cos(dLon);
      return (toDeg(Math.atan2(y, x)) + 360) % 360;
    }
    
    // Rota noktaları oluştur
    const allPorts = [
      { name: departurePort, lat: departureLat, lon: departureLon, isPort: true },
      ...(intermediatePorts || []).map((p: any) => ({ name: p.name, lat: p.lat, lon: p.lon, isPort: true })),
      { name: destinationPort, lat: destinationLat, lon: destinationLon, isPort: true },
    ];
    
    const speed = plannedSpeed || 12; // varsayılan 12 knot
    const waypoints: any[] = [];
    let totalDistance = 0;
    let totalHours = 0;
    let wpSequence = 0;
    const passedStraits: any[] = [];
    let ecaZoneHours = 0;
    
    for (let i = 0; i < allPorts.length; i++) {
      const port = allPorts[i];
      
      // Port waypoint ekle
      waypoints.push({
        sequence: wpSequence++,
        name: port.name,
        latitude: port.lat,
        longitude: port.lon,
        waypointType: "port",
        isPort: true,
        speedKnots: 0,
        distanceToNextNm: 0,
        courseToNext: 0,
        legTimeHours: 0,
        eta: null,
        etd: null,
      });
      
      // Sonraki port'a git
      if (i < allPorts.length - 1) {
        const nextPort = allPorts[i + 1];
        const legDistance = haversineNm(port.lat, port.lon, nextPort.lat, nextPort.lon);
        
        // Bu leg'de hangi boğazlar/kanallar var?
        for (const strait of STRAITS_AND_CANALS) {
          const distToStrait = haversineNm(port.lat, port.lon, strait.lat, strait.lon);
          const distStraitToNext = haversineNm(strait.lat, strait.lon, nextPort.lat, nextPort.lon);
          
          // Eğer boğaz, kalkış ve varış arasındaysa (toplam mesafe %20'den fazla sapmıyorsa)
          if (distToStrait + distStraitToNext < legDistance * 1.2 && distToStrait > 10 && distStraitToNext > 10) {
            passedStraits.push(strait);
            waypoints.push({
              sequence: wpSequence++,
              name: strait.name,
              latitude: strait.lat,
              longitude: strait.lon,
              waypointType: strait.type,
              isStrait: strait.type === "strait",
              isCanal: strait.type === "canal",
              isPort: false,
              speedKnots: strait.type === "canal" ? 8 : speed,
              distanceToNextNm: 0, // sonra hesaplanacak
              courseToNext: 0,
              legTimeHours: strait.transitTimeHours,
              eta: null,
              etd: null,
              notes: `Transit time: ~${strait.transitTimeHours}h`,
            });
          }
        }
      }
    }
    
    // Waypoint'ları sırala ve mesafe/kurs hesapla
    waypoints.sort((a: any, b: any) => {
      if (a.isPort && !b.isPort) return -1; // portlar önce... aslında sequence'a göre
      return a.sequence - b.sequence;
    });
    
    // Tekrar sırala: portlar ve boğazlar arası mesafeleri hesapla
    for (let i = 0; i < waypoints.length - 1; i++) {
      const wp = waypoints[i];
      const next = waypoints[i + 1];
      if (wp.latitude && wp.longitude && next.latitude && next.longitude) {
        wp.distanceToNextNm = Math.round(haversineNm(wp.latitude, wp.longitude, next.latitude, next.longitude) * 10) / 10;
        wp.courseToNext = Math.round(bearing(wp.latitude, wp.longitude, next.latitude, next.longitude) * 10) / 10;
        const wpSpeed = wp.speedKnots || speed;
        wp.legTimeHours = wp.isStrait || wp.isCanal ? wp.legTimeHours : Math.round((wp.distanceToNextNm / wpSpeed) * 10) / 10;
        totalDistance += wp.distanceToNextNm;
        totalHours += wp.legTimeHours;
      }
      
      // ECA zone kontrolü
      for (const eca of ECA_ZONES) {
        if (wp.latitude >= eca.latMin && wp.latitude <= eca.latMax && wp.longitude >= eca.lonMin && wp.longitude <= eca.lonMax) {
          wp.isEcaZone = true;
          ecaZoneHours += wp.legTimeHours || 0;
          break;
        }
      }
    }
    
    // ETA hesapla
    const departureTime = req.body.departureDate ? new Date(req.body.departureDate) : new Date();
    let currentTime = new Date(departureTime);
    for (const wp of waypoints) {
      wp.eta = new Date(currentTime);
      if (wp.isPort && wp !== waypoints[0]) {
        // Port'ta kalış süresi (varsayılan 0, ara limanlar için ayarlanabilir)
        wp.portStayHours = wp.portStayHours || 0;
      }
      currentTime = new Date(currentTime.getTime() + (wp.legTimeHours || 0) * 3600000 + (wp.portStayHours || 0) * 3600000);
      wp.etd = new Date(currentTime);
    }
    
    const totalDays = Math.round((totalHours / 24) * 10) / 10;
    const ecaZoneDays = Math.round((ecaZoneHours / 24) * 10) / 10;
    
    res.json({
      waypoints,
      summary: {
        totalDistanceNm: Math.round(totalDistance * 10) / 10,
        totalHours: Math.round(totalHours * 10) / 10,
        totalDays,
        plannedSpeed: speed,
        straitsAndCanals: passedStraits.map(s => ({ name: s.name, type: s.type, transitTimeHours: s.transitTimeHours })),
        ecaZoneDays,
        arrivalDate: waypoints[waypoints.length - 1]?.etd || null,
      },
    });
  } catch (e: any) {
    console.error("[passage-plan] Route calculation error:", e.message);
    res.status(500).json({ error: e.message });
  }
});

// POST /calculate-fuel — Yakıt hesaplama
router.post("/calculate-fuel", isAuthenticated, async (req: any, res) => {
  try {
    const { totalDays, ecaZoneDays, plannedSpeed, vesselType, deadweight, grossTonnage } = req.body;
    
    // Basit yakıt tüketimi tahmini (ton/gün)
    // Gerçek projede gemi motor bilgisine göre hesaplanır
    const dwt = deadweight || 50000;
    const gt = grossTonnage || 30000;
    
    // Yaklaşık günlük tüketim (DWT ve hıza göre)
    let dailyHfo: number;
    let dailyMgo: number;
    
    if (dwt < 10000) {
      // Small vessel
      dailyHfo = 8 + (plannedSpeed - 10) * 1.5;
      dailyMgo = 1.5;
    } else if (dwt < 30000) {
      // Handysize
      dailyHfo = 15 + (plannedSpeed - 12) * 2.5;
      dailyMgo = 2.5;
    } else if (dwt < 60000) {
      // Supramax/Panamax
      dailyHfo = 25 + (plannedSpeed - 13) * 3.5;
      dailyMgo = 3.5;
    } else if (dwt < 120000) {
      // Capesize
      dailyHfo = 40 + (plannedSpeed - 13) * 5;
      dailyMgo = 4;
    } else {
      // VLCC/ULCC
      dailyHfo = 55 + (plannedSpeed - 14) * 7;
      dailyMgo = 5;
    }
    
    dailyHfo = Math.max(dailyHfo, 5); // minimum
    dailyMgo = Math.max(dailyMgo, 1);
    
    const nonEcaDays = (totalDays || 0) - (ecaZoneDays || 0);
    const ecaDays = ecaZoneDays || 0;
    
    // ECA bölgelerinde MGO kullanılır, HFO kullanılmaz
    const totalHfo = Math.round(nonEcaDays * dailyHfo * 10) / 10;
    const totalMgo = Math.round((nonEcaDays * dailyMgo + ecaDays * (dailyHfo + dailyMgo)) * 10) / 10;
    const totalFuel = Math.round((totalHfo + totalMgo) * 10) / 10;
    
    // Maliyet tahmini (yaklaşık fiyatlar)
    const hfoPrice = 450; // USD/ton
    const mgoPrice = 750; // USD/ton
    const estimatedCost = Math.round(totalHfo * hfoPrice + totalMgo * mgoPrice);
    
    res.json({
      dailyHfoConsumption: Math.round(dailyHfo * 10) / 10,
      dailyMgoConsumption: Math.round(dailyMgo * 10) / 10,
      totalHfoConsumption: totalHfo,
      totalMgoConsumption: totalMgo,
      totalFuelConsumption: totalFuel,
      ecaZoneDays: ecaDays,
      nonEcaDays: Math.round(nonEcaDays * 10) / 10,
      estimatedFuelCost: estimatedCost,
      fuelPrices: { hfo: hfoPrice, mgo: mgoPrice },
    });
  } catch (e: any) {
    console.error("[passage-plan] Fuel calculation error:", e.message);
    res.status(500).json({ error: e.message });
  }
});
