const BASE = 'https://api.datalastic.com/api/v0';
const apiKey = () => process.env.DATALASTIC_API_KEY;

// Helper
async function datalasticGet(endpoint: string, params: Record<string, string>): Promise<any> {
  const key = apiKey();
  if (!key) return null;
  
  try {
    const queryParams = new URLSearchParams({ 'api-key': key, ...params });
    const url = `${BASE}/${endpoint}?${queryParams.toString()}`;
    const response = await fetch(url);
    const data = await response.json();
    
    if (data.meta?.success) return data.data;
    console.log(`Datalastic ${endpoint} failed:`, data);
    return null;
  } catch (error) {
    console.error(`Datalastic ${endpoint} error:`, error);
    return null;
  }
}

// 1. SHIP TRACKING — Gerçek zamanlı pozisyon
export async function getVesselPosition(params: { imo?: string; mmsi?: string }) {
  const p: any = {};
  if (params.imo) p.imo = params.imo;
  if (params.mmsi) p.mmsi = params.mmsi;
  return datalasticGet('vessel', p);
}

  if (params.imo) p.imo = params.imo;
  if (params.mmsi) p.mmsi = params.mmsi;
  return datalasticGet('vessel_pro', p);
}

// 3. BULK TRACKING — 100 gemiye kadar toplu
export async function getVesselBulk(imoList: string[]) {
  const key = apiKey();
  if (!key || imoList.length === 0) return [];
  try {
    const url = `${BASE}/vessel_bulk?api-key=${key}&imo=${imoList.join(',')}`;
    const response = await fetch(url);
    const data = await response.json();
    if (data.meta?.success) return Array.isArray(data.data) ? data.data : [data.data];
    return [];
  } catch { return []; }
}

// 5. VESSEL FINDER — İsim/tip/ülke ile gemi ara
export async function findVessels(params: { name?: string; type?: string; country_iso?: string }) {
  const p: any = {};
  if (params.name) p.name = params.name;
  if (params.type) p.type = params.type;
  if (params.country_iso) p.country_iso = params.country_iso;
  const result = await datalasticGet('vessel_find', p);
  return result?.results || [];
}

// 6. HISTORICAL TRACKING — Geçmiş pozisyonlar
export async function getVesselHistory(params: { imo?: string; mmsi?: string; days?: number }) {

  if (params.mmsi) p.mmsi = params.mmsi;
  if (params.days) p.days = String(params.days);
  return datalasticGet('vessel_history', p);
}

// 7. LOCATION TRACKING — Belirli alandaki gemiler (radius)
export async function getVesselsInRadius(params: { port_unlocode?: string; lat?: number; lon?: number; radius?: number }) {
  const p: any = {};

  if (params.lat) p.lat = String(params.lat);
  if (params.lon) p.lon = String(params.lon);
  p.radius = String(params.radius || 5);
  const result = await datalasticGet('vessel_inradius', p);
  return Array.isArray(result) ? result : result ? [result] : [];
}

=== 5. CACHE STRATEJİSİ ===

Kredi tasarrufu için agresif cache:
- Pozisyon (vessel, fleet): 1 dakika (short)
- Port traffic: 5 dakika (medium)
- Gemi geçmişi: 1 saat (long)
- Gemi info/specs: 24 saat (daily)
