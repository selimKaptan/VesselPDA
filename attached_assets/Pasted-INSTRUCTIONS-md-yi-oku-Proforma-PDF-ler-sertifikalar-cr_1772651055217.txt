1. server/file-storage.ts dosyasını güncelle veya oluştur:

   import fs from 'fs';
   import path from 'path';
   import crypto from 'crypto';

   const UPLOAD_DIR = path.join(process.cwd(), 'uploads');
   
   // Alt klasörler
   const DIRS = {
     documents: path.join(UPLOAD_DIR, 'documents'),
     certificates: path.join(UPLOAD_DIR, 'certificates'),
     crew: path.join(UPLOAD_DIR, 'crew'),
     proformas: path.join(UPLOAD_DIR, 'proformas'),
     logos: path.join(UPLOAD_DIR, 'logos'),
     bids: path.join(UPLOAD_DIR, 'bids'),
   };

   // Klasörleri oluştur
   Object.values(DIRS).forEach(dir => {
     if (!fs.existsSync(dir)) fs.mkdirSync(dir, { recursive: true });
   });

   export function saveBase64File(base64Data: string, category: keyof typeof DIRS, originalName?: string): string {
     // data:application/pdf;base64,xxxx formatını parse et
     const matches = base64Data.match(/^data:(.+);base64,(.+)$/);
     const buffer = Buffer.from(matches ? matches[2] : base64Data, 'base64');
     const ext = matches ? matches[1].split('/')[1] : 'bin';
     const fileName = `${crypto.randomUUID()}.${ext}`;
     const filePath = path.join(DIRS[category], fileName);
     fs.writeFileSync(filePath, buffer);
     return `/uploads/${category}/${fileName}`;
   }

   export function getFileAsBase64(filePath: string): string | null {
     const fullPath = path.join(process.cwd(), filePath);
     if (!fs.existsSync(fullPath)) return null;
     const buffer = fs.readFileSync(fullPath);
     const ext = path.extname(filePath).slice(1);
     const mimeTypes: Record<string, string> = {
       pdf: 'application/pdf', png: 'image/png', jpg: 'image/jpeg',
       jpeg: 'image/jpeg', doc: 'application/msword', docx: 'application/vnd.openxmlformats-officedocument.wordprocessingml.document'
     };
     const mime = mimeTypes[ext] || 'application/octet-stream';
     return `data:${mime};base64,${buffer.toString('base64')}`;
   }

   export function deleteFile(filePath: string): boolean {
     const fullPath = path.join(process.cwd(), filePath);
     if (fs.existsSync(fullPath)) { fs.unlinkSync(fullPath); return true; }
     return false;
   }
