## 2. BACKEND — Cargo Parcels API

server/routes/voyage.routes.ts dosyasına ekle:

// GET /voyages/:id/cargo-parcels
router.get("/:id/cargo-parcels", isAuthenticated, async (req: any, res) => {
  const voyageId = parseInt(req.params.id);
  const parcels = await db.select().from(cargoParcels)
    .where(eq(cargoParcels.voyageId, voyageId))
    .orderBy(asc(cargoParcels.sortOrder));
  const [stowage] = await db.select().from(stowagePlans)
    .where(eq(stowagePlans.voyageId, voyageId))
    .orderBy(desc(stowagePlans.createdAt))
    .limit(1);
  
  const totalTarget = parcels.reduce((s, p) => s + (p.targetQuantity || 0), 0);
  const totalHandled = parcels.reduce((s, p) => s + (p.handledQuantity || 0), 0);
  const progress = totalTarget > 0 ? Math.round((totalHandled / totalTarget) * 1000) / 10 : 0;
  
  res.json({
    parcels,
    stowagePlan: stowage || null,
    summary: { totalTarget, totalHandled, remaining: totalTarget - totalHandled, progress, parcelCount: parcels.length },
  });
});

// POST /voyages/:id/cargo-parcels
router.post("/:id/cargo-parcels", isAuthenticated, async (req: any, res) => {
  const [parcel] = await db.insert(cargoParcels).values({
    ...req.body,
    voyageId: parseInt(req.params.id),
  }).returning();
  res.status(201).json(parcel);
});

// PATCH /cargo-parcels/:pid
router.patch("/cargo-parcels/:pid", isAuthenticated, async (req: any, res) => {
  const [parcel] = await db.update(cargoParcels).set({ ...req.body, updatedAt: new Date() })
    .where(eq(cargoParcels.id, parseInt(req.params.pid))).returning();
  res.json(parcel);
});

// DELETE /cargo-parcels/:pid
router.delete("/cargo-parcels/:pid", isAuthenticated, async (req: any, res) => {
  await db.delete(cargoParcels).where(eq(cargoParcels.id, parseInt(req.params.pid)));
  res.json({ ok: true });
});

// PATCH /voyages/:id/stowage-plan
router.patch("/:id/stowage-plan", isAuthenticated, async (req: any, res) => {
  const voyageId = parseInt(req.params.id);
  const existing = await db.select().from(stowagePlans).where(eq(stowagePlans.voyageId, voyageId)).limit(1);
  if (existing.length > 0) {
    const [plan] = await db.update(stowagePlans).set(req.body).where(eq(stowagePlans.id, existing[0].id)).returning();
    res.json(plan);
  } else {
    const [plan] = await db.insert(stowagePlans).values({ ...req.body, voyageId }).returning();
    res.json(plan);
  }
});
