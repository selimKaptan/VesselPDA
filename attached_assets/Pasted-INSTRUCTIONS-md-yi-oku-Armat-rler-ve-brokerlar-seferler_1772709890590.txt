// Voyage Davetleri
export const voyageInvitations = pgTable("voyage_invitations", {
  id: integer("id").primaryKey().generatedAlwaysAsIdentity(),
  voyageId: integer("voyage_id").notNull().references(() => voyages.id, { onDelete: "cascade" }),
  inviterId: varchar("inviter_id").notNull().references(() => users.id),
  inviteeId: varchar("invitee_id").references(() => users.id), // platform kullanıcısı ise
  inviteeEmail: varchar("invitee_email", { length: 300 }), // platform dışı ise
  inviteeCompanyId: integer("invitee_company_id").references(() => companyProfiles.id),
  inviteeRole: varchar("invitee_role", { length: 30 }).notNull(), // agent, provider, surveyor, broker, observer
  serviceType: varchar("service_type", { length: 100 }), // stevedoring, surveying, fumigation, customs, etc.
  message: text("message"), // davet mesajı
  status: varchar("status", { length: 20 }).default("pending"), // pending, accepted, declined, expired, cancelled
  token: varchar("token", { length: 100 }), // email ile davet için
  respondedAt: timestamp("responded_at"),
  declineReason: text("decline_reason"),
  expiresAt: timestamp("expires_at"),
  createdAt: timestamp("created_at").defaultNow(),
});

// Voyage Katılımcıları (davet kabul edilince buraya eklenir)
export const voyageParticipants = pgTable("voyage_participants", {
  id: integer("id").primaryKey().generatedAlwaysAsIdentity(),
  voyageId: integer("voyage_id").notNull().references(() => voyages.id, { onDelete: "cascade" }),
  userId: varchar("user_id").notNull().references(() => users.id),
  companyProfileId: integer("company_profile_id").references(() => companyProfiles.id),
  role: varchar("role", { length: 30 }).notNull(), // agent, provider, surveyor, broker, observer
  serviceType: varchar("service_type", { length: 100 }),
  permissions: jsonb("permissions").$type<VoyageParticipantPermissions>().default({}),
  isActive: boolean("is_active").default(true),
  joinedAt: timestamp("joined_at").defaultNow(),
});

export interface VoyageParticipantPermissions {
  canViewDocuments?: boolean;
  canUploadDocuments?: boolean;
  canChat?: boolean;
  canViewFinancials?: boolean;
  canEditChecklist?: boolean;
  canViewSOF?: boolean;
}

// Default permissions by role:
// agent: tümü true
// provider: viewDocs, uploadDocs, chat true. Financials false.
// surveyor: viewDocs, uploadDocs, chat true.
// broker: viewDocs, chat, viewFinancials true.
// observer: sadece viewDocs, viewSOF true.

export const voyageInviteRelations = relations(voyageInvitations, ({ one }) => ({
  voyage: one(voyages, { fields: [voyageInvitations.voyageId], references: [voyages.id] }),
  inviter: one(users, { fields: [voyageInvitations.inviterId], references: [users.id] }),
  invitee: one(users, { fields: [voyageInvitations.inviteeId], references: [users.id] }),
  inviteeCompany: one(companyProfiles, { fields: [voyageInvitations.inviteeCompanyId], references: [companyProfiles.id] }),
}));

export const voyageParticipantRelations = relations(voyageParticipants, ({ one }) => ({
  voyage: one(voyages, { fields: [voyageParticipants.voyageId], references: [voyages.id] }),
  user: one(users, { fields: [voyageParticipants.userId], references: [users.id] }),
  companyProfile: one(companyProfiles, { fields: [voyageParticipants.companyProfileId], references: [companyProfiles.id] }),
}));

export const insertVoyageInviteSchema = createInsertSchema(voyageInvitations).omit({ id: true, createdAt: true });
export const insertVoyageParticipantSchema = createInsertSchema(voyageParticipants).omit({ id: true, joinedAt: true });

=== 2. API ENDPOINTS ===

server/routes/voyage-invite.routes.ts oluştur:

// Voyage'daki davetleri listele
GET /api/voyages/:voyageId/invitations
- isAuthenticated
- Voyage'a ait tüm davetler
- inviter ve invitee bilgileri ile join
- Sadece voyage owner, agent veya katılımcılar görebilir

// Davet gönder
POST /api/voyages/:voyageId/invite
- isAuthenticated
- Body: {
    inviteeId?: string,       // platform kullanıcısı (opsiyonel)
    inviteeEmail?: string,    // email ile davet (opsiyonel)
    inviteeCompanyId?: number, // şirket profili (opsiyonel)
    inviteeRole: string,      // agent, provider, surveyor, broker, observer
    serviceType?: string,     // stevedoring, surveying, etc.
    message?: string,         // kişisel davet mesajı
  }
- Kontroller:
  * Sadece voyage owner veya agent gönderebilir
  * Aynı kişiye aynı voyage için tekrar davet gönderemez
  * inviteeId VEYA inviteeEmail zorunlu (en az biri)
- Token üret (email davet için)
- ExpiresAt: 7 gün sonra
- Platform kullanıcısına: notification + email
- Platform dışına: sadece email
- Voyage activity log: "Invitation sent to {name} as {role}"

// Toplu davet
POST /api/voyages/:voyageId/invite-bulk
- Body: { invitations: [{ email, role, serviceType }...] }
- Birden fazla kişiyi aynı anda davet et
- Her biri için ayrı davet kaydı oluştur

// Daveti kabul et (platform kullanıcısı)
POST /api/voyages/:voyageId/invitations/:inviteId/accept
- isAuthenticated
- Sadece davet edilen kişi kabul edebilir
- Status: accepted, respondedAt: now()
- voyageParticipants tablosuna ekle (role ve permissions ile)
- Voyage chat'e otomatik "welcome" mesajı
- Notification: inviter'a "{name} accepted your invitation"
- Voyage activity: "{name} joined as {role}"

// Daveti reddet
POST /api/voyages/:voyageId/invitations/:inviteId/decline
- isAuthenticated
- Body: { reason?: string }
- Status: declined, declineReason
- Notification: inviter'a "{name} declined the invitation"

// Email token ile kabul (platform dışı)
POST /api/voyage-invites/:token/accept
- Public endpoint (token auth)
- Token valid mi, expired mı kontrol
- Kullanıcı kayıtlıysa: kabul et, participant ekle
- Kayıtlı değilse: register sayfasına yönlendir, kayıt sonrası otomatik kabul

// Daveti iptal et
DELETE /api/voyages/:voyageId/invitations/:inviteId
- Sadece inviter veya voyage owner
- Status: cancelled

// Daveti tekrar gönder
POST /api/voyages/:voyageId/invitations/:inviteId/resend
- Yeni token, yeni expiry
- Email tekrar gönder

// Voyage katılımcıları
GET /api/voyages/:voyageId/participants
- Tüm aktif katılımcılar
- User + company bilgileri ile
- Role ve permissions

// Katılımcı güncelle
PATCH /api/voyages/:voyageId/participants/:participantId
- Role veya permissions güncelle
- Sadece voyage owner

// Katılımcıyı çıkar
DELETE /api/voyages/:voyageId/participants/:participantId
- Katılımcıyı deaktive et
- Sadece voyage owner veya kendisi (ayrılma)
- Voyage activity: "{name} left the voyage"

// Benim davetlerim
GET /api/my-voyage-invitations
- isAuthenticated
- Kullanıcıya gelen tüm bekleyen voyage davetleri
- Badge count için de kullanılır

server/routes.ts'e mount et:
import voyageInviteRoutes from './routes/voyage-invite.routes';
app.use('/api', voyageInviteRoutes);
