// Gerçek zamanlı gemi pozisyonu
export async function getVesselPosition(params: { imo?: string; mmsi?: string }): Promise<DatalasticVesselPosition | null> {
  const apiKey = process.env.DATALASTIC_API_KEY;
  if (!apiKey) {
    console.log('Datalastic: No API key, using mock data');
    return null;
  }
  
  try {
    const queryParam = params.imo ? `imo=${params.imo}` : `mmsi=${params.mmsi}`;
    const url = `${DATALASTIC_BASE}/vessel?api-key=${apiKey}&${queryParam}`;
    
    const response = await fetch(url);
    const data = await response.json();
    
    if (data.meta?.success && data.data) {
      return data.data;
    }
    return null;
  } catch (error) {
    console.error('Datalastic vessel position error:', error);
    return null;
  }
}

export async function getVesselsInPort(params: { portUnlocode?: string; lat?: number; lon?: number; radius?: number }): Promise<any[]> {

    } else {

// Liman bilgisi
export async function findPort(name: string): Promise<any[]> {
  const apiKey = process.env.DATALASTIC_API_KEY;
  if (!apiKey) return [];
  
  try {
    const url = `${DATALASTIC_BASE}/port_find?api-key=${apiKey}&name=${encodeURIComponent(name)}`;
    
    const response = await fetch(url);
    const data = await response.json();
    
    if (data.meta?.success && data.data) return Array.isArray(data.data) ? data.data : [data.data];
    return [];
  } catch (error) {
    console.error('Datalastic port find error:', error);
    return [];
  }
}

// Gemi tiplerine göre ara
export async function findVessels(params: { type?: string; country?: string; name?: string }): Promise<any[]> {
  const apiKey = process.env.DATALASTIC_API_KEY;
  if (!apiKey) return [];
  
  try {
    const queryParams = new URLSearchParams();
    queryParams.set('api-key', apiKey);
    if (params.type) queryParams.set('type', params.type);
    if (params.country) queryParams.set('country_iso', params.country);
    if (params.name) queryParams.set('name', params.name);
    
    const url = `${DATALASTIC_BASE}/vessel_find?${queryParams.toString()}`;
    
    const response = await fetch(url);
    const data = await response.json();
    
    if (data.meta?.success && data.data?.results) return data.data.results;
    return [];
  } catch (error) {
    console.error('Datalastic find error:', error);
    return [];
  }
}

// API durumu ve kalan kredi kontrolü
export function isDatalasticConfigured(): boolean {
  return !!process.env.DATALASTIC_API_KEY;
}

=== 6. CACHE STRATEJİSİ (Kredi Tasarrufu) ===

Datalastic kredileri değerli — agresif cache kullan:

- Tek gemi pozisyonu: 1 dakika cache (short)
- Fleet toplu pozisyon: 1 dakika cache (short)
- Gemi geçmişi: 1 saat cache (long)
- Gemi info (statik): 24 saat cache (daily)
- Port traffic: 5 dakika cache (medium)
- IMO lookup: 24 saat cache (daily)

Bu sayede aynı gemiyi 60 kez/saat sorgulasanız bile sadece 1 kredi harcar.

=== 7. FALLBACK SİSTEMİ ===

Her fonksiyonda 3 katman:
1. Cache'de varsa → cache'den dön (0 kredi)
2. Datalastic API varsa → API'den çek (1 kredi)
3. Mevcut AIS Stream varsa → oradan çek
4. Hiçbiri yoksa → mock data

Bu sayede API key olmasa bile platform çalışmaya devam eder.

=== 8. IMPORTANT ===

- DATALASTIC_API_KEY env variable'dan okunacak
- API key yoksa CRASH ETMEMELİ — mock data'ya düşmeli
- Cache agresif kullan — her istek kredi harcar
- Bulk endpoint kullan (100 gemi = 1 istek vs 100 istek)
- Rate limit: Datalastic'in kendi limiti var, ek limit gerekmez
- Mevcut AIS Stream entegrasyonunu SILME — fallback olarak kalsın
- Mevcut mock data da kalsın — son çare fallback
- vessel_pro endpoint'i daha pahalı ama ETA/draft veriyor — fleet için basic, detay için pro kullan
- Mevcut tracking routes yapısını bozma — üzerine ekle/güncelle
- Frontend'de data source badge göster (Datalastic/AIS/Mock)
