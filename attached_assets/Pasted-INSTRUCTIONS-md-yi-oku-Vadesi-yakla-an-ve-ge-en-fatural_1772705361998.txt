function reminderTemplate(params: {
  recipientName: string;
  invoiceTitle: string;
  amount: number;
  currency: string;
  dueDate: string;
  daysUntilDue: number;
  invoiceUrl: string;
}): { subject: string; html: string } {
  // email-templates.ts'deki baseTemplate'i kullan
  // Sarı/amber tonunda "Payment Reminder" temalı email
  // "Your invoice is due in X days" mesajı
  // Büyük "Pay Now" butonu

}

function overdueTemplate(params: {
  recipientName: string;
  invoiceTitle: string;
  amount: number;
  currency: string;
  dueDate: string;
  daysOverdue: number;
  invoiceUrl: string;
}): { subject: string; html: string } {

export async function checkAndSendReminders() {
  const now = new Date();
  const threeDaysFromNow = new Date(now.getTime() + 3 * 24 * 60 * 60 * 1000);

  try {
    // 1. Vadesi 3 gün içinde olan, henüz hatırlatma gönderilmemiş faturalar
    const upcomingInvoices = await db.select().from(invoices).where(
      and(
        eq(invoices.status, 'pending'),
        lte(invoices.dueDate, threeDaysFromNow),
        gt(invoices.dueDate, now),
        isNull(invoices.reminderSentAt)
      )
    );

    for (const invoice of upcomingInvoices) {
      if (!invoice.recipientEmail) continue;
      
      const daysUntilDue = Math.ceil((new Date(invoice.dueDate).getTime() - now.getTime()) / (1000 * 60 * 60 * 24));
      
      // Email gönder
      const { subject, html } = reminderTemplate({
        recipientName: invoice.recipientName || 'Dear Sir/Madam',
        invoiceTitle: invoice.title,
        amount: Number(invoice.amount),
        currency: invoice.currency || 'USD',
        dueDate: new Date(invoice.dueDate).toLocaleDateString('en-GB'),
        daysUntilDue,
        invoiceUrl: `https://vesselpda.com/invoices`,
      });

      try {
        await sendEmail({ to: invoice.recipientEmail, subject, html });
        await db.update(invoices).set({ reminderSentAt: now }).where(eq(invoices.id, invoice.id));
        console.log(`Reminder sent for invoice ${invoice.id}`);
      } catch (e) {
        console.error(`Failed to send reminder for invoice ${invoice.id}:`, e);
      }
    }

    // 2. Vadesi geçmiş, henüz overdue hatırlatma gönderilmemiş faturalar
    const overdueInvoices = await db.select().from(invoices).where(
      and(
        eq(invoices.status, 'pending'),
        lt(invoices.dueDate, now),
        isNull(invoices.overdueReminderSentAt)
      )
    );

    for (const invoice of overdueInvoices) {
      if (!invoice.recipientEmail) continue;
      
      const daysOverdue = Math.ceil((now.getTime() - new Date(invoice.dueDate).getTime()) / (1000 * 60 * 60 * 24));
      
      const { subject, html } = overdueTemplate({
        recipientName: invoice.recipientName || 'Dear Sir/Madam',
        invoiceTitle: invoice.title,
        amount: Number(invoice.amount),
        currency: invoice.currency || 'USD',
        dueDate: new Date(invoice.dueDate).toLocaleDateString('en-GB'),
        daysOverdue,
        invoiceUrl: `https://vesselpda.com/invoices`,
      });

      try {
        await sendEmail({ to: invoice.recipientEmail, subject, html });
        await db.update(invoices).set({ overdueReminderSentAt: now }).where(eq(invoices.id, invoice.id));
        
        // Platform notification da oluştur
        await storage.createNotification({
          userId: invoice.userId,
          type: 'invoice_overdue',
          title: 'Invoice Overdue',
          message: `Invoice "${invoice.title}" is ${daysOverdue} days overdue`,
          link: '/invoices',
        });
        
        console.log(`Overdue reminder sent for invoice ${invoice.id}`);
      } catch (e) {
        console.error(`Failed to send overdue reminder for invoice ${invoice.id}:`, e);
      }
    }

    // 3. 7 gün geçmiş overdue'lara tekrar hatırlatma (haftalık)
    const weeklyOverdue = new Date(now.getTime() - 7 * 24 * 60 * 60 * 1000);
    const longOverdueInvoices = await db.select().from(invoices).where(
      and(
        eq(invoices.status, 'pending'),
        lt(invoices.dueDate, now),
        lt(invoices.overdueReminderSentAt, weeklyOverdue)
      )
    );

    for (const invoice of longOverdueInvoices) {
      if (!invoice.recipientEmail) continue;
      const daysOverdue = Math.ceil((now.getTime() - new Date(invoice.dueDate).getTime()) / (1000 * 60 * 60 * 24));
      
      const { subject, html } = overdueTemplate({
        recipientName: invoice.recipientName || 'Dear Sir/Madam',
        invoiceTitle: invoice.title,
        amount: Number(invoice.amount),
        currency: invoice.currency || 'USD',
        dueDate: new Date(invoice.dueDate).toLocaleDateString('en-GB'),
        daysOverdue,
        invoiceUrl: `https://vesselpda.com/invoices`,
      });

      try {
        await sendEmail({ to: invoice.recipientEmail, subject, html });
        await db.update(invoices).set({ overdueReminderSentAt: now }).where(eq(invoices.id, invoice.id));
      } catch (e) {
        console.error(`Failed to send weekly overdue reminder:`, e);
      }
    }

    return {
      upcomingReminders: upcomingInvoices.length,
      overdueReminders: overdueInvoices.length,
      weeklyReminders: longOverdueInvoices.length,
    };
  } catch (error) {
    console.error('Payment reminder check failed:', error);
    return { error: error.message };
  }
}

// sendEmail helper — mevcut Resend altyapısını kullan
async function sendEmail(params: { to: string; subject: string; html: string }) {
  // server/email.ts'deki mevcut resend instance'ını kullan
  const { Resend } = require('resend');
  const resend = new Resend(process.env.RESEND_API_KEY);
  await resend.emails.send({
    from: 'VesselPDA <noreply@vesselpda.com>',
    to: params.to,
    subject: params.subject,
    html: params.html,
  });
}

=== 3. CRON JOB / SCHEDULED CHECK ===

server/index.ts'e ekle:

import { checkAndSendReminders } from './payment-reminders';

// Her gün sabah 09:00'da hatırlatma kontrolü
// setInterval ile basit cron (production'da proper cron job kullanılabilir)
function startReminderScheduler() {
  // İlk çalıştırma: 1 dakika sonra
  setTimeout(async () => {
    console.log('Running initial payment reminder check...');
    const result = await checkAndSendReminders();
    console.log('Reminder check result:', result);
  }, 60000);

  // Sonra her 24 saatte bir
  setInterval(async () => {
    console.log('Running scheduled payment reminder check...');
    const result = await checkAndSendReminders();
    console.log('Reminder check result:', result);
  }, 24 * 60 * 60 * 1000);
}

// Server başlatıldıktan sonra çağır:
startReminderScheduler();

=== 7. IMPORTANT ===

- Email hatırlatma sadece recipientEmail dolu olan faturalara gönderilir
- Aynı faturaya günde 1'den fazla hatırlatma gitmez
- Overdue: haftalık tekrar (7 günde bir)
- Ödenen faturalara (status: paid) hatırlatma gitmez
- İptal edilen faturalara (status: cancelled) hatırlatma gitmez
- setInterval basit çözüm — production'da node-cron veya external scheduler kullanılabilir
- Email gönderme hatası ana servisi durdurmaz (try-catch)
