export function getVoyageFeatures(operationType?: string | null) {
  const type = (operationType || "other").toLowerCase().replace(/[\s-]/g, "_") as VoyageOperationType;
  return VOYAGE_FEATURES[type] || VOYAGE_FEATURES.other;
}

// Timeline step ikonlarını belirle
export function getTimelineIcon(step: string): string {
  const icons: Record<string, string> = {
    "Pre-Arrival": "navigation",
    "Anchorage": "anchor",
    "Berthed": "ship",
    "Cargo Ops": "package",
    "Crew Ops": "users",
    "Bunkering": "fuel",
    "Docking": "wrench",
    "Repairs": "tool",
    "Undocking": "ship",
    "Departed": "navigation-2",
    "Departure": "navigation",
    "Transit": "compass",
    "Arrival": "map-pin",
    "Operations": "activity",
  };
  return icons[step] || "circle";
}

## 4. VOYAGE OLUŞTURMADA OPERASYOn TİPİ SEÇİMİ

Yeni voyage oluşturma formunda "Operation Type" alanı zaten var ise kontrol et, yoksa ekle:
- Select/dropdown: Loading, Discharging, Crew Change, Bunkering, Transit, Drydock, Other
- Bu seçim voyage.operationType veya voyage.purposeOfCall alanına kaydedilsin
- DB'deki mevcut voyage'ların operationType alanını kontrol et — boşsa purposeOfCall'dan çıkar

Migration gerekmez — mevcut schema'daki operationType veya purposeOfCall alanını kullan.
