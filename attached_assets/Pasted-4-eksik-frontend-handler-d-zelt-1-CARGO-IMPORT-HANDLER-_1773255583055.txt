## 1. CARGO IMPORT HANDLER EKSİK — Özet beyan parse ediliyor ama parcels kaydedilmiyor

client/src/pages/voyage-detail.tsx dosyasında declaration modal'ın import butonunun handler fonksiyonunu bul veya ekle:

Eğer handleImportParcels veya handleImportAll fonksiyonu yoksa ekle:

async function handleImportParcels() {
  try {
    let imported = 0;
    for (const [i, parcel] of parsedParcels.entries()) {
      const res = await apiRequest("POST", `/api/v1/voyages/${voyageId}/cargo-parcels`, {
        receiverName: parcel.receiverName,
        shipperName: parcel.shipperName,
        cargoType: parcel.cargoType,
        cargoGrade: parcel.cargoGrade,
        targetQuantity: parcel.quantity,
        blQuantity: parcel.quantity,
        unit: parcel.unit || "MT",
        blNumber: parcel.blNumber,
        holdNumbers: parcel.holdNumbers,
        tankNumbers: parcel.tankNumbers,
        dischargeSequence: i + 1,
        loadingSequence: i + 1,
        notes: [
          parcel.hsCode ? `HS: ${parcel.hsCode}` : "",
          parcel.countryOfOrigin ? `Origin: ${parcel.countryOfOrigin}` : "",
          parcel.portOfLoading ? `POL: ${parcel.portOfLoading}` : "",
        ].filter(Boolean).join(" · ") || null,
      });
      if (res.ok) imported++;
    }
    toast({ title: "Cargo Imported", description: `${imported} parcels created from declaration` });
    setShowDeclarationModal(false);
    setParsedParcels([]);
    queryClient.invalidateQueries();
  } catch (e: any) {
    toast({ title: "Error", description: e.message, variant: "destructive" });
  }
}

Bu fonksiyonu declaration modal'ın "Import X Parcels" butonunun onClick'ine bağla:
<Button onClick={handleImportParcels} disabled={parsedParcels.length === 0}>
  Import {parsedParcels.length} Parcels
</Button>
