### D) STATUS QUICK-TOGGLE — Kart üzerinde tek tıkla durum değiştir

Kart header'daki status badge'e tıklayınca sırayla: waiting → active → paused → completed

function cycleParcelStatus(parcelId: number) {
  const parcel = parcels.find(p => p.id === parcelId);
  const sequence = ["waiting", "active", "paused", "completed"];
  const currentIdx = sequence.indexOf(parcel?.operationStatus || "waiting");
  const nextStatus = sequence[(currentIdx + 1) % sequence.length];
  
  const updates: any = { operationStatus: nextStatus };
  if (nextStatus === "active" && !parcel?.operationStartedAt) {
    updates.operationStartedAt = new Date().toISOString();
  }
  if (nextStatus === "completed") {
    updates.operationCompletedAt = new Date().toISOString();
  }
  
  updateParcel(parcelId, updates);
}

const sortedParcels = useMemo(() => {
  const sorted = [...parcels];
  switch (sortBy) {
    case "sequence": return sorted.sort((a, b) => (a.dischargeSequence || 99) - (b.dischargeSequence || 99));
    case "status": return sorted.sort((a, b) => {
      const order = { active: 0, paused: 1, waiting: 2, completed: 3 };
      return (order[a.operationStatus] || 9) - (order[b.operationStatus] || 9);
    });
    case "quantity": return sorted.sort((a, b) => (b.targetQuantity || 0) - (a.targetQuantity || 0));
    case "progress": return sorted.sort((a, b) => {
      const pA = a.targetQuantity > 0 ? (a.handledQuantity || 0) / a.targetQuantity : 0;
      const pB = b.targetQuantity > 0 ? (b.handledQuantity || 0) / b.targetQuantity : 0;
      return pB - pA;
    });
    default: return sorted;
  }
}, [parcels, sortBy]);
