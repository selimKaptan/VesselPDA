## 1. LIMAN ADI FORMATTER FONKSİYONU

client/src/lib/format-port.ts dosyası oluştur:

// AIS destination formatlarını temizle
// Örnekler: "IT GOA > TR IZM", "TRIZM", "TR IZM", "IZMIR ALSANCAK", ">>>MERSIN"
export function formatPortName(raw?: string | null): string | null {
  if (!raw || raw.trim() === "" || raw === "null" || raw === "undefined") return null;
  
  let name = raw.trim();
  
  // ">>>" prefix temizle
  name = name.replace(/^>+/, "").trim();
  
  // "XX XXX > YY YYY" formatı — ">" sonrasını al (varış)
  if (name.includes(">")) {
    const parts = name.split(">");
    name = parts[parts.length - 1].trim();
  }
  
  // 2 harfli ülke kodu + boşluk + liman kodu (Örn: "TR IZM") — ülke kodunu kaldır
  if (/^[A-Z]{2}\s[A-Z]{3,}$/.test(name)) {
    name = name.substring(3);
  }
  
  // LOCODE formatı "TRIZM" → sadece IZM kısmını al (5 karakter, hepsi büyük harf)
  if (/^[A-Z]{5}$/.test(name)) {
    name = name.substring(2);
  }
  
  // Title Case'e çevir
  return name
    .toLowerCase()
    .split(/[\s_-]+/)
    .map(word => word.charAt(0).toUpperCase() + word.slice(1))
    .join(" ");
}

// Kalkış limanını belirle — birden fazla kaynaktan
export function getDeparturePort(voyage: any, vesselData: any): string | null {
  // 1. Voyage'dan
  if (voyage?.departurePort) return formatPortName(voyage.departurePort);
  if (voyage?.loadPort) return formatPortName(voyage.loadPort);
  
  // 2. Datalastic vessel_pro'dan — last_port
  if (vesselData?.last_port_name) return formatPortName(vesselData.last_port_name);
  if (vesselData?.last_port?.name) return formatPortName(vesselData.last_port.name);
  
  // 3. AIS'den
  if (vesselData?.ais?.last_port) return formatPortName(vesselData.ais.last_port);
  
  return null;
}

// Varış limanını belirle — birden fazla kaynaktan
export function getDestinationPort(voyage: any, vesselData: any): string | null {
  // 1. Voyage'dan
  if (voyage?.portName) return formatPortName(voyage.portName);
  if (voyage?.dischargePort) return formatPortName(voyage.dischargePort);
  if (voyage?.destinationPort) return formatPortName(voyage.destinationPort);
  
  // 2. Datalastic'ten — AIS destination
  if (vesselData?.destination) return formatPortName(vesselData.destination);
  if (vesselData?.ais?.destination) return formatPortName(vesselData.ais.destination);
  if (vesselData?.next_port_name) return formatPortName(vesselData.next_port_name);
  
  return null;
}

## 4. KOORDİNATLARI FORMATLAMA

Enlem/Boylam değerlerini daha okunabilir yap:

ÖNCE: 38.82209 / 26.94614
SONRA: 38°49.3'N / 26°56.8'E

function formatCoord(value: number, type: "lat" | "lon"): string {
  const abs = Math.abs(value);
  const deg = Math.floor(abs);
  const min = ((abs - deg) * 60).toFixed(1);
  const dir = type === "lat" ? (value >= 0 ? "N" : "S") : (value >= 0 ? "E" : "W");
  return `${deg}°${min}'${dir}`;
}
