  // documentType values:

  // ballast_water — Ballast Water Management Certificate

  // radio — Radio Station License

// Geminin tüm vault belgeleri
GET /api/vessels/:vesselId/vault
- isAuthenticated
- Tüm aktif belgeler (archived hariç)
- Category ve documentType'a göre grupla
- Her belgenin expiry durumunu hesapla:
  * expired: expiryDate < now → status: expired (kırmızı)
  * expiring_soon: expiryDate < now + 30 gün → status: expiring_soon (amber)
  * active: diğer → status: active (yeşil)
  * no_expiry: expiryDate null → permanent (mavi)

Response:
{
  documents: [...],
  summary: {
    total: 15,
    active: 10,
    expiringSoon: 3,
    expired: 2,
    completeness: 78, // zorunlu belge yüzdesi
  },
  requiredDocuments: [
    // Eksik zorunlu belgeler listesi
    { documentType: "ballast_water", name: "Ballast Water Management Certificate", missing: true },
  ]
}

export async function autoAttachVaultDocuments(voyageId: number, vesselId: number) {
  // 1. Geminin vault'undaki autoAttachToVoyage=true ve status=active olan belgeleri çek
  const vaultDocs = await db.select().from(vesselDocumentVault).where(
    and(
      eq(vesselDocumentVault.vesselId, vesselId),
      eq(vesselDocumentVault.autoAttachToVoyage, true),
      eq(vesselDocumentVault.status, 'active'),
    )
  );

  // 2. Her belgeyi voyage'a iliştir
  for (const doc of vaultDocs) {
    await db.insert(voyageVaultDocuments).values({
      voyageId,
      vaultDocumentId: doc.id,
      attachedBy: 'auto',
    });
  }

  // 3. Voyage activity log
  if (vaultDocs.length > 0) {
    await logVoyageActivity({
      voyageId,
      activityType: 'documents_auto_attached',
      title: `${vaultDocs.length} documents auto-attached from vessel vault`,
      description: vaultDocs.map(d => d.documentName).join(', '),
    });
  }

  return vaultDocs.length;
}

import { db } from './db';
import { vesselDocumentVault, vessels, users } from '../shared/schema';
import { and, eq, lt, gt, lte } from 'drizzle-orm';

export async function checkVaultExpiryReminders() {
  const now = new Date();
  const in30Days = new Date(now.getTime() + 30 * 24 * 60 * 60 * 1000);
  const in14Days = new Date(now.getTime() + 14 * 24 * 60 * 60 * 1000);
  const in7Days = new Date(now.getTime() + 7 * 24 * 60 * 60 * 1000);

  let totalSent = 0;

  // 30 gün kala uyarı
  const docs30 = await db.select({
    doc: vesselDocumentVault,
    vesselName: vessels.name,
    userEmail: users.email,
    userName: users.firstName,
  })
  .from(vesselDocumentVault)
  .innerJoin(vessels, eq(vesselDocumentVault.vesselId, vessels.id))
  .innerJoin(users, eq(vesselDocumentVault.userId, users.id))
  .where(and(
    eq(vesselDocumentVault.status, 'active'),
    lte(vesselDocumentVault.expiryDate, in30Days),
    gt(vesselDocumentVault.expiryDate, in14Days),
    eq(vesselDocumentVault.reminderSent30, false),
  ));

  for (const row of docs30) {
    const daysLeft = Math.ceil((new Date(row.doc.expiryDate).getTime() - now.getTime()) / (1000 * 60 * 60 * 24));
    
    // Email gönder
    const { subject, html } = documentExpiryTemplate({
      userName: row.userName || 'Dear Sir/Madam',
      vesselName: row.vesselName,
      documentName: row.doc.documentName,
      documentType: row.doc.documentType,
      expiryDate: new Date(row.doc.expiryDate).toLocaleDateString('en-GB'),
      daysLeft,
      severity: 'warning', // 30 gün: warning (amber)
      vaultUrl: `https://vesselpda.com/vessel-vault/${row.doc.vesselId}`,
    });

    try {
      if (row.userEmail) await sendEmail({ to: row.userEmail, subject, html });
      
      // Platform notification
      await storage.createNotification({
        userId: row.doc.userId,
        type: 'document_expiring',
        title: `Document Expiring in ${daysLeft} days`,
        message: `${row.doc.documentName} for ${row.vesselName} expires on ${new Date(row.doc.expiryDate).toLocaleDateString('en-GB')}`,
        link: `/vessel-vault/${row.doc.vesselId}`,
      });

      await db.update(vesselDocumentVault).set({ reminderSent30: true }).where(eq(vesselDocumentVault.id, row.doc.id));
      totalSent++;
    } catch (e) {
      console.error('Vault reminder error:', e);
    }
  }

  // 14 gün kala uyarı (aynı pattern, reminderSent14)
  const docs14 = await db.select({
    doc: vesselDocumentVault,
    vesselName: vessels.name,
    userEmail: users.email,
    userName: users.firstName,
  })
  .from(vesselDocumentVault)
  .innerJoin(vessels, eq(vesselDocumentVault.vesselId, vessels.id))
  .innerJoin(users, eq(vesselDocumentVault.userId, users.id))
  .where(and(
    eq(vesselDocumentVault.status, 'active'),
    lte(vesselDocumentVault.expiryDate, in14Days),
    gt(vesselDocumentVault.expiryDate, in7Days),
    eq(vesselDocumentVault.reminderSent14, false),
  ));

  for (const row of docs14) {
    const daysLeft = Math.ceil((new Date(row.doc.expiryDate).getTime() - now.getTime()) / (1000 * 60 * 60 * 24));
    const { subject, html } = documentExpiryTemplate({
      userName: row.userName || 'Dear Sir/Madam',
      vesselName: row.vesselName,
      documentName: row.doc.documentName,
      documentType: row.doc.documentType,
      expiryDate: new Date(row.doc.expiryDate).toLocaleDateString('en-GB'),
      daysLeft,
      severity: 'urgent', // 14 gün: urgent (orange)
      vaultUrl: `https://vesselpda.com/vessel-vault/${row.doc.vesselId}`,
    });
    try {
      if (row.userEmail) await sendEmail({ to: row.userEmail, subject, html });
      await storage.createNotification({ userId: row.doc.userId, type: 'document_expiring_urgent', title: `URGENT: Document Expiring in ${daysLeft} days`, message: `${row.doc.documentName} for ${row.vesselName}`, link: `/vessel-vault/${row.doc.vesselId}` });
      await db.update(vesselDocumentVault).set({ reminderSent14: true }).where(eq(vesselDocumentVault.id, row.doc.id));
      totalSent++;
    } catch (e) { console.error(e); }
  }

  // 7 gün kala uyarı
  const docs7 = await db.select({
    doc: vesselDocumentVault,
    vesselName: vessels.name,
    userEmail: users.email,
    userName: users.firstName,
  })
  .from(vesselDocumentVault)
  .innerJoin(vessels, eq(vesselDocumentVault.vesselId, vessels.id))
  .innerJoin(users, eq(vesselDocumentVault.userId, users.id))
  .where(and(
    eq(vesselDocumentVault.status, 'active'),
    lte(vesselDocumentVault.expiryDate, in7Days),
    gt(vesselDocumentVault.expiryDate, now),
    eq(vesselDocumentVault.reminderSent7, false),
  ));

  for (const row of docs7) {
    const daysLeft = Math.ceil((new Date(row.doc.expiryDate).getTime() - now.getTime()) / (1000 * 60 * 60 * 24));
    const { subject, html } = documentExpiryTemplate({
      userName: row.userName || 'Dear Sir/Madam',
      vesselName: row.vesselName,
      documentName: row.doc.documentName,
      documentType: row.doc.documentType,
      expiryDate: new Date(row.doc.expiryDate).toLocaleDateString('en-GB'),
      daysLeft,
      severity: 'critical', // 7 gün: critical (kırmızı)
      vaultUrl: `https://vesselpda.com/vessel-vault/${row.doc.vesselId}`,
    });
    try {
      if (row.userEmail) await sendEmail({ to: row.userEmail, subject, html });
      await storage.createNotification({ userId: row.doc.userId, type: 'document_expiring_critical', title: `CRITICAL: ${row.doc.documentName} expires in ${daysLeft} days!`, message: `${row.vesselName} — Renew immediately`, link: `/vessel-vault/${row.doc.vesselId}` });
      await db.update(vesselDocumentVault).set({ reminderSent7: true }).where(eq(vesselDocumentVault.id, row.doc.id));
      totalSent++;
    } catch (e) { console.error(e); }
  }

  // Süresi dolmuş belgeleri güncelle
  const expired = await db.select().from(vesselDocumentVault).where(and(
    eq(vesselDocumentVault.status, 'active'),
    lt(vesselDocumentVault.expiryDate, now),
  ));
  for (const doc of expired) {
    await db.update(vesselDocumentVault).set({ status: 'expired' }).where(eq(vesselDocumentVault.id, doc.id));
    if (!doc.reminderSentExpired) {
      // Expired notification
      await storage.createNotification({ userId: doc.userId, type: 'document_expired', title: `EXPIRED: ${doc.documentName}`, message: `This document has expired. Please upload renewed version.`, link: `/vessel-vault/${doc.vesselId}` });
      await db.update(vesselDocumentVault).set({ reminderSentExpired: true }).where(eq(vesselDocumentVault.id, doc.id));
    }
  }

  return { remindersSent: totalSent, expiredUpdated: expired.length };
}

export function documentExpiryTemplate(params: {
  userName: string;
  vesselName: string;
  documentName: string;
  documentType: string;
  expiryDate: string;
  daysLeft: number;
  severity: 'warning' | 'urgent' | 'critical';
  vaultUrl: string;
}): { subject: string; html: string } {
  // severity bazlı renk ve mesaj:
  // warning (30 gün): amber tema, "Expiring Soon"
  // urgent (14 gün): orange tema, "Expiring — Action Required"
  // critical (7 gün): kırmızı tema, "EXPIRING — Immediate Action Required"
  
  // baseTemplate kullan
  // İçerik:
  // Büyük uyarı ikonu ve başlık
  // Vessel ve belge bilgileri tablosu
  // "Expires in X days" büyük countdown gösterge
  // "Upload Renewed Document" butonu → vaultUrl
  // Alt not: "Renew this document to maintain compliance"
  
  const severityConfig = {
    warning: { color: '#F59E0B', label: 'Expiring Soon', icon: '⚠️' },
    urgent: { color: '#F97316', label: 'Action Required', icon: '🔶' },
    critical: { color: '#EF4444', label: 'Immediate Action Required', icon: '🔴' },
  };
  const config = severityConfig[params.severity];
  
  return {
    subject: `${config.icon} ${params.documentName} — ${config.label} (${params.daysLeft} days) — ${params.vesselName}`,
    html: baseTemplate(`...email content with ${config.color} theme...`),
  };
}

=== 10. IMPORTANT ===

- Vault belgeleri gemiye ait, kullanıcıya değil — gemi el değiştirse belgeler kalır
- Auto-attach SADECE active + autoAttachToVoyage=true belgeleri iliştirir
- Expired belgeler otomatik iliştirilmez
- Yeni versiyon yüklenince eski: superseded, yeni: active olur
- Voyage'a iliştirme referans bazlı — dosya kopyalanmaz, vault'a link
- Reminder sadece 1 kez gönderilir (per threshold)
- Belge silinince voyage bağlantıları da temizlenir
- File upload mevcut file-storage.ts altyapısını kullansın
- PDF, JPEG, PNG, DOC, DOCX desteklenmeli
- Max dosya boyutu: 10MB
- Mevcut vesselCertificates tablosu ile çakışma olmasın — vault ayrı, daha kapsamlı
