ENTEGRASYON A — PDA → FDA otomatik aktarım:
server/routes/fda.routes.ts dosyasında FDA oluşturma endpoint'ini bul. Eğer proformaId varsa PDA verilerini çek:

if (req.body.proformaId) {
  const proforma = await storage.getProformaById(req.body.proformaId);
  if (proforma && proforma.lineItems) {
    const fdaLineItems = (proforma.lineItems as any[]).map((item, idx) => ({
      id: `fda-${idx + 1}`,
      description: item.description,
      category: item.category || "other",
      estimatedUsd: item.amountUsd || 0,
      estimatedEur: item.amountEur || 0,
      actualUsd: 0,
      actualEur: 0,
      varianceUsd: -(item.amountUsd || 0),
      variancePercent: -100,
      remarks: "",
    }));
    req.body.lineItems = fdaLineItems;
    req.body.totalEstimatedUsd = proforma.totalUsd;
    req.body.totalEstimatedEur = proforma.totalEur || 0;
    req.body.totalActualUsd = 0;
    req.body.totalActualEur = 0;
    req.body.vesselName = req.body.vesselName || proforma.vessel?.name || "";
    req.body.portName = req.body.portName || proforma.port?.name || "";
    req.body.exchangeRate = proforma.exchangeRate || null;
    req.body.bankDetails = proforma.bankDetails || null;
  }
}

ENTEGRASYON B — Port Expense → FDA senkron:
server/routes/port-expense.routes.ts dosyasında yeni expense oluşturulduktan sonra:

if (expense.fdaId) {
  try {
    const [fda] = await db.select().from(fdaAccounts).where(eq(fdaAccounts.id, expense.fdaId));
    if (fda) {
      const expenses = await storage.getPortExpensesByFda(expense.fdaId);
      const totalActualUsd = expenses.reduce((sum, e) => sum + (e.amountUsd || e.amount || 0), 0);
      const varianceUsd = totalActualUsd - (fda.totalEstimatedUsd || 0);
      const variancePercent = fda.totalEstimatedUsd ? (varianceUsd / fda.totalEstimatedUsd) * 100 : 0;
      await db.update(fdaAccounts).set({
        totalActualUsd,
        varianceUsd,
        variancePercent: Math.round(variancePercent * 10) / 10,
        updatedAt: new Date(),
      }).where(eq(fdaAccounts.id, expense.fdaId));
      if (Math.abs(variancePercent) > 10) {
        await storage.createNotification({
          userId: fda.userId,
          type: "fda_variance_alert",
          title: "FDA Variance Alert",
          message: `FDA ${fda.referenceNumber}: Actual expenses deviate ${variancePercent.toFixed(1)}% from PDA estimate.`,
          link: `/fda/${fda.id}`,
        });
      }
    }
  } catch (err) { console.error("[port-expense] FDA sync failed:", err); }
}

if (!expense.fdaId && expense.voyageId) {

ENTEGRASYON C — FDA → Invoice + DA Advance mutabakat:
server/routes/fda.routes.ts dosyasında FDA status "approved" yapıldığında:

if (updatedFda.status === "approved") {
  try {
    const invoiceTitle = `FDA Settlement — ${updatedFda.referenceNumber || ""}`;
    const invoiceAmount = updatedFda.totalActualUsd || 0;
    const existingInvoice = await db.select().from(invoices)
      .where(eq(invoices.fdaId, updatedFda.id))
      .limit(1);
    if (existingInvoice.length === 0 && invoiceAmount > 0) {
      let totalAdvanceReceived = 0;
      if (updatedFda.voyageId) {
        const advances = await storage.getDaAdvancesByVoyage(updatedFda.voyageId);
        totalAdvanceReceived = advances.reduce((sum, a) => sum + (a.receivedAmount || 0), 0);
      }
      const netAmount = invoiceAmount - totalAdvanceReceived;
      const invoice = await storage.createInvoice({
        voyageId: updatedFda.voyageId || null,
        fdaId: updatedFda.id,
        proformaId: updatedFda.proformaId || null,
        createdByUserId: updatedFda.userId,
        title: invoiceTitle,
        amount: Math.abs(netAmount),
        currency: "USD",
        dueDate: new Date(Date.now() + 30 * 24 * 60 * 60 * 1000),
        invoiceType: netAmount >= 0 ? "invoice" : "credit_note",
        notes: totalAdvanceReceived > 0
          ? `FDA Total: $${invoiceAmount.toLocaleString()} | DA Advance Received: $${totalAdvanceReceived.toLocaleString()} | Net: $${netAmount.toLocaleString()}`
          : null,
      });
      console.log(`[fda] Auto-created invoice #${invoice.id} from FDA #${updatedFda.id} (net: $${netAmount})`);
      await storage.createNotification({
        userId: updatedFda.userId,
        type: "invoice_created",
        title: "Invoice Auto-Created",
        message: `Invoice "${invoiceTitle}" for $${Math.abs(netAmount).toLocaleString()} has been auto-created from approved FDA.`,
        link: `/invoices`,
      });
    }
  } catch (err) { console.error("[fda] Auto-invoice creation failed:", err); }
}
