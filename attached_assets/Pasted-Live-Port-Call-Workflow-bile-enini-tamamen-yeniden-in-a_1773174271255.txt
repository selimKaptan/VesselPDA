// Operasyon tipine göre görünecek adımları filtrele
export function getStepsForOperation(operationType: string): WorkflowStep[] {
  const type = operationType?.toLowerCase().replace(/[\s-]/g, "_") || "all";
  return PORT_CALL_STEPS.filter(step => 
    step.applicableTo.includes("all") || step.applicableTo.includes(type as any)
  );
}

// Deadline hesapla
export function calculateDeadline(step: WorkflowStep, previousStepTime?: Date | null): Date | null {
  if (!step.hasDeadline || !step.deadlineHoursAfterPrevious || !previousStepTime) return null;
  return new Date(previousStepTime.getTime() + step.deadlineHoursAfterPrevious * 3600000);
}

// Rolün iç notları görebilme yetkisi
export function canSeeInternalNotes(role: string): boolean {
  return ["admin", "agency", "agent", "superintendent"].includes(role?.toLowerCase());
}

export function canEditWorkflow(role: string): boolean {
  return ["admin", "agency", "agent"].includes(role?.toLowerCase());
}

## 3. BACKEND — Workflow Step'leri Kaydet/Güncelle

server/routes/voyage.routes.ts dosyasına ekle:

// POST /voyages/:id/workflow-step — Adım tamamla
router.post("/:id/workflow-step", isAuthenticated, async (req: any, res) => {
  const voyageId = parseInt(req.params.id);
  const { stepKey, completedAt, notes } = req.body;
  const userId = req.user?.id;
  const userName = req.user?.firstName ? `${req.user.firstName} ${req.user.lastName || ""}`.trim() : "Unknown";
  
  // portCallWorkflowSteps tablosu yoksa, voyage'ın metadata/jsonb alanına kaydet
  // Veya mevcut port_call_events tablosunu kullan
  
  try {
    // Voyage'ın workflowSteps JSON alanını güncelle
    const [voyage] = await db.select().from(voyages).where(eq(voyages.id, voyageId));
    if (!voyage) return res.status(404).json({ message: "Voyage not found" });
    
    const currentSteps = (voyage.workflowSteps as Record<string, any>) || {};
    currentSteps[stepKey] = {
      completedAt,
      completedBy: userName,
      completedByUserId: userId,
      notes: notes || null,
      updatedAt: new Date().toISOString(),
    };
    
    await db.update(voyages).set({ 
      workflowSteps: currentSteps,
      updatedAt: new Date(),
    }).where(eq(voyages.id, voyageId));
    
    res.json({ ok: true, steps: currentSteps });
  } catch (e: any) {
    res.status(500).json({ error: e.message });
  }
});

// GET /voyages/:id/workflow — Workflow durumunu getir
router.get("/:id/workflow", isAuthenticated, async (req: any, res) => {
  const voyageId = parseInt(req.params.id);
  const [voyage] = await db.select().from(voyages).where(eq(voyages.id, voyageId));
  if (!voyage) return res.status(404).json({ message: "Voyage not found" });
  res.json({
    operationType: voyage.operationType || voyage.purposeOfCall || "loading",
    steps: voyage.workflowSteps || {},
  });
});
