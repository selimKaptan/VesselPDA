=== 1. VOYAGE ACTIVITY LOG TABLOSU ===

shared/schema.ts'e ekle:

export const voyageActivities = pgTable("voyage_activities", {
  id: integer("id").primaryKey().generatedAlwaysAsIdentity(),
  voyageId: integer("voyage_id").notNull().references(() => voyages.id, { onDelete: "cascade" }),
  userId: varchar("user_id").references(() => users.id),
  activityType: varchar("activity_type", { length: 50 }).notNull(),
  title: varchar("title", { length: 300 }).notNull(),
  description: text("description"),
  metadata: jsonb("metadata"),
  createdAt: timestamp("created_at").defaultNow(),
});

// activityType values:
// voyage_created, status_changed, eta_updated, etd_updated,
// document_uploaded, document_signed, checklist_added, checklist_completed,
// chat_message, sof_created, sof_finalized, pda_created, pda_approved,
// fda_created, fda_approved, invoice_created, invoice_paid,
// nomination_sent, nomination_accepted, appointment_created,
// service_request_created, service_offer_selected, custom_note

export const voyageActivityRelations = relations(voyageActivities, ({ one }) => ({
  voyage: one(voyages, { fields: [voyageActivities.voyageId], references: [voyages.id] }),
  user: one(users, { fields: [voyageActivities.userId], references: [users.id] }),
}));

export const insertVoyageActivitySchema = createInsertSchema(voyageActivities).omit({ id: true, createdAt: true });

=== 2. ACTIVITY LOGGING HELPER ===

server/voyage-activity.ts oluştur:

import { db } from './db';
import { voyageActivities } from '../shared/schema';

export async function logVoyageActivity(params: {
  voyageId: number;
  userId?: string;
  activityType: string;
  title: string;
  description?: string;
  metadata?: any;
}) {
  try {
    await db.insert(voyageActivities).values(params);
  } catch (e) {
    console.error('Failed to log voyage activity:', e);
  }
}

=== 3. ACTIVITY LOG ENTEGRASYONU ===

Mevcut voyage ile ilgili TÜM endpoint'lerde activity log ekle:

voyage.routes.ts:
- POST /api/voyages → logVoyageActivity({ voyageId, userId, activityType: 'voyage_created', title: 'Voyage created' })
- PATCH /api/voyages/:id → status değiştiyse: activityType: 'status_changed', title: `Status changed to ${newStatus}`
- PATCH /api/voyages/:id → ETA değiştiyse: activityType: 'eta_updated', title: `ETA updated to ${newEta}`
- POST /api/voyages/:id/checklist → activityType: 'checklist_added', title: `Checklist item added: ${title}`
- PATCH /api/voyages/:id/checklist/:itemId → isCompleted true ise: activityType: 'checklist_completed', title: `Completed: ${title}`
- POST /api/voyages/:id/documents → activityType: 'document_uploaded', title: `Document uploaded: ${name}`
- POST /api/voyages/:id/documents/:docId/sign → activityType: 'document_signed', title: `Document signed: ${name}`
- POST /api/voyages/:id/chat → activityType: 'chat_message', title: `Message from ${userName}`
- POST /api/voyages/:id/reviews → activityType: 'review_submitted', title: `Review submitted: ${rating}/5`

sof.routes.ts:
- POST /api/sof (eğer voyageId varsa) → activityType: 'sof_created', title: 'Statement of Facts created'
- POST /api/sof/:id/finalize → activityType: 'sof_finalized', title: 'SOF finalized'

proforma.routes.ts:
- POST /api/proformas (eğer voyageId ile bağlıysa) → activityType: 'pda_created', title: `PDA created: ${refNumber}`
- POST /api/proformas/:id/approve → activityType: 'pda_approved', title: 'PDA approved'

fda.routes.ts:
- POST /api/fda (eğer voyageId varsa) → activityType: 'fda_created', title: `FDA created: ${refNumber}`
- POST /api/fda/:id/approve → activityType: 'fda_approved', title: 'FDA approved'

invoice routes:
- POST /api/invoices (eğer voyageId varsa) → activityType: 'invoice_created', title: `Invoice created: ${amount} ${currency}`
- PATCH /api/invoices/:id/pay → activityType: 'invoice_paid', title: 'Invoice paid'

nomination routes:
- POST /api/tenders/:id/nominate → activityType: 'nomination_sent', title: 'Agent nominated'

=== 4. ACTIVITY TIMELINE API ===

GET /api/voyages/:id/activities
- isAuthenticated
- voyageId'ye göre tüm aktiviteleri çek
- createdAt DESC sırala
- User bilgisi ile join (firstName, lastName)
- Response: { activities: [...], total: number }

GET /api/user/recent-activity?limit=20
- isAuthenticated
- Kullanıcının tüm voyage'larındaki son aktiviteleri çek
- Veya kullanıcının yaptığı tüm aktiviteleri çek
- Dashboard'da gösterilecek
