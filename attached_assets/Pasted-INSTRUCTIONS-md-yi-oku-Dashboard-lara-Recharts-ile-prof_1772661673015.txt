   // Aylık trend verileri (tüm roller için)
   app.get("/api/stats/trends", isAuthenticated, async (req: any, res: any, next: any) => {
     try {
       const userId = req.user.claims.sub;
       const user = await storage.getUser(userId);
       const months = parseInt(req.query.months as string) || 6;
       
       // Son X ay için aylık istatistikler
       const { rows: proformaTrend } = await pool.query(`
         SELECT 
           TO_CHAR(created_at, 'YYYY-MM') as month,
           COUNT(*) as count,
           COALESCE(SUM(total_usd::numeric), 0) as total_usd
         FROM proformas 
         WHERE created_at > NOW() - INTERVAL '${months} months'
         ${user?.userRole !== 'admin' ? `AND user_id = '${userId}'` : ''}
         GROUP BY TO_CHAR(created_at, 'YYYY-MM')
         ORDER BY month
       `);

       const { rows: tenderTrend } = await pool.query(`
         SELECT 
           TO_CHAR(created_at, 'YYYY-MM') as month,
           COUNT(*) as count,
           COUNT(*) FILTER (WHERE status = 'nominated') as completed
         FROM port_tenders 
         WHERE created_at > NOW() - INTERVAL '${months} months'
         ${user?.userRole !== 'admin' ? `AND user_id = '${userId}'` : ''}
         GROUP BY TO_CHAR(created_at, 'YYYY-MM')
         ORDER BY month
       `);

       const { rows: voyageTrend } = await pool.query(`
         SELECT 
           TO_CHAR(created_at, 'YYYY-MM') as month,
           COUNT(*) as count,
           COUNT(*) FILTER (WHERE status = 'completed') as completed
         FROM voyages 
         WHERE created_at > NOW() - INTERVAL '${months} months'
         ${user?.userRole !== 'admin' ? `AND (user_id = '${userId}' OR agent_user_id = '${userId}')` : ''}
         GROUP BY TO_CHAR(created_at, 'YYYY-MM')
         ORDER BY month
       `);

       res.json({ proformaTrend, tenderTrend, voyageTrend });
     } catch (error) { next(error); }
   });

   // Rol bazlı dashboard stats
   app.get("/api/stats/dashboard", isAuthenticated, async (req: any, res: any, next: any) => {
     try {
       const userId = req.user.claims.sub;
       const user = await storage.getUser(userId);
       const role = user?.userRole;

       let stats: any = {};

       if (role === 'admin') {
         const { rows } = await pool.query(`
           SELECT
             (SELECT COUNT(*) FROM users WHERE is_suspended = false) as total_users,
             (SELECT COUNT(*) FROM users WHERE created_at > NOW() - INTERVAL '30 days') as new_users_month,
             (SELECT COUNT(*) FROM proformas) as total_proformas,
             (SELECT COUNT(*) FROM proformas WHERE created_at > NOW() - INTERVAL '30 days') as proformas_month,
             (SELECT COUNT(*) FROM port_tenders WHERE status = 'open') as open_tenders,
             (SELECT COUNT(*) FROM voyages WHERE status = 'in_progress') as active_voyages,
             (SELECT COUNT(*) FROM company_profiles WHERE is_approved = true) as verified_companies,
             (SELECT COALESCE(SUM(total_usd::numeric), 0) FROM proformas WHERE created_at > NOW() - INTERVAL '30 days') as revenue_month
         `);
         stats = rows[0];
       } else if (role === 'shipowner' || role === 'broker') {
         const { rows } = await pool.query(`
           SELECT
             (SELECT COUNT(*) FROM vessels WHERE user_id = $1) as my_vessels,
             (SELECT COUNT(*) FROM port_tenders WHERE user_id = $1 AND status = 'open') as my_open_tenders,
             (SELECT COUNT(*) FROM port_tenders WHERE user_id = $1 AND status = 'nominated') as my_completed_tenders,
             (SELECT COUNT(*) FROM voyages WHERE user_id = $1 AND status = 'in_progress') as my_active_voyages,
             (SELECT COUNT(*) FROM proformas WHERE user_id = $1) as my_proformas,
             (SELECT COALESCE(SUM(total_usd::numeric), 0) FROM proformas WHERE user_id = $1) as my_total_spend
         `, [userId]);
         stats = rows[0];
       } else if (role === 'agent') {
         const { rows } = await pool.query(`
           SELECT
             (SELECT COUNT(*) FROM tender_bids WHERE agent_user_id = $1) as my_bids,
             (SELECT COUNT(*) FROM tender_bids WHERE agent_user_id = $1 AND status = 'selected') as won_bids,
             (SELECT COUNT(*) FROM voyages WHERE agent_user_id = $1 AND status = 'in_progress') as active_voyages,
             (SELECT COUNT(*) FROM voyages WHERE agent_user_id = $1 AND status = 'completed') as completed_voyages,
             (SELECT COUNT(*) FROM direct_nominations WHERE agent_user_id = $1 AND status = 'pending') as pending_nominations,
             (SELECT COALESCE(AVG(rating), 0) FROM agent_reviews ar JOIN company_profiles cp ON ar.company_profile_id = cp.id WHERE cp.user_id = $1) as avg_rating
         `, [userId]);
         stats = rows[0];
       } else if (role === 'provider') {
         const { rows } = await pool.query(`
           SELECT
             (SELECT COUNT(*) FROM service_offers WHERE provider_user_id = $1) as my_offers,
             (SELECT COUNT(*) FROM service_offers WHERE provider_user_id = $1 AND status = 'selected') as won_offers,
             (SELECT COUNT(*) FROM service_requests WHERE status = 'open') as open_requests
         `, [userId]);
         stats = rows[0];
       }

       // Tender status dağılımı (pie chart için)
       const { rows: tenderDistribution } = await pool.query(`
         SELECT status, COUNT(*) as count
         FROM port_tenders
         ${role !== 'admin' ? `WHERE user_id = '${userId}'` : ''}
         GROUP BY status
       `);

       // Voyage status dağılımı
       const { rows: voyageDistribution } = await pool.query(`
         SELECT status, COUNT(*) as count
         FROM voyages
         ${role !== 'admin' ? `WHERE user_id = '${userId}' OR agent_user_id = '${userId}'` : ''}
         GROUP BY status
       `);

       res.json({ stats, tenderDistribution, voyageDistribution });
     } catch (error) { next(error); }
   });

2. Yeni dashboard chart component oluştur: client/src/components/dashboards/dashboard-charts.tsx

   import { useQuery } from '@tanstack/react-query';
   import {
     LineChart, Line, BarChart, Bar, PieChart, Pie, Cell,
     XAxis, YAxis, CartesianGrid, Tooltip, Legend, ResponsiveContainer, Area, AreaChart
   } from 'recharts';
   import { Card } from '@/components/ui/card';
   import { Skeleton } from '@/components/ui/skeleton';

   const COLORS = ['#38BDF8', '#22C55E', '#F59E0B', '#EF4444', '#8B5CF6', '#EC4899'];

   // Ay ismini kısalt: 2025-01 → Jan
   function formatMonth(ym: string) {
     const months = ['Jan','Feb','Mar','Apr','May','Jun','Jul','Aug','Sep','Oct','Nov','Dec'];
     const m = parseInt(ym.split('-')[1]) - 1;
     return months[m] || ym;
   }

   export function ProformaTrendChart() {
     const { data, isLoading } = useQuery({
       queryKey: ['/api/stats/trends?months=6'],
     });

     if (isLoading) return <Skeleton className="h-64 w-full" />;
     const chartData = (data?.proformaTrend || []).map((d: any) => ({
       month: formatMonth(d.month),
       count: Number(d.count),
       totalUsd: Math.round(Number(d.total_usd)),
     }));

     return (
       <Card className="p-4">
         <h3 className="text-sm font-semibold text-slate-300 mb-3">Proforma Trend (6 Months)</h3>
         <ResponsiveContainer width="100%" height={220}>
           <AreaChart data={chartData}>
             <defs>
               <linearGradient id="colorProforma" x1="0" y1="0" x2="0" y2="1">
                 <stop offset="5%" stopColor="#38BDF8" stopOpacity={0.3}/>
                 <stop offset="95%" stopColor="#38BDF8" stopOpacity={0}/>
               </linearGradient>
             </defs>
             <CartesianGrid strokeDasharray="3 3" stroke="#1e293b" />
             <XAxis dataKey="month" stroke="#64748b" fontSize={12} />
             <YAxis stroke="#64748b" fontSize={12} />
             <Tooltip
               contentStyle={{ backgroundColor: '#1e293b', border: '1px solid #334155', borderRadius: '8px' }}
               labelStyle={{ color: '#94a3b8' }}
               itemStyle={{ color: '#f1f5f9' }}
             />
             <Area type="monotone" dataKey="count" stroke="#38BDF8" fill="url(#colorProforma)" name="Proformas" />
           </AreaChart>
         </ResponsiveContainer>
       </Card>
     );
   }

   export function TenderTrendChart() {
     const { data, isLoading } = useQuery({

   export function StatusDistributionChart({ title, data }: { title: string; data: any[] }) {
     if (!data || data.length === 0) return null;
     const chartData = data.map((d: any) => ({
       name: d.status || 'unknown',
       value: Number(d.count),
     }));

     return (
       <Card className="p-4">
         <h3 className="text-sm font-semibold text-slate-300 mb-3">{title}</h3>
         <ResponsiveContainer width="100%" height={200}>
           <PieChart>
             <Pie
               data={chartData}
               cx="50%"
               cy="50%"
               innerRadius={50}
               outerRadius={80}
               paddingAngle={3}
               dataKey="value"
               label={({ name, percent }) => `${name} ${(percent * 100).toFixed(0)}%`}
             >
               {chartData.map((_, index) => (
                 <Cell key={index} fill={COLORS[index % COLORS.length]} />
               ))}
             </Pie>
             <Tooltip
               contentStyle={{ backgroundColor: '#1e293b', border: '1px solid #334155', borderRadius: '8px' }}
               itemStyle={{ color: '#f1f5f9' }}
             />
           </PieChart>
         </ResponsiveContainer>
       </Card>
     );
   }

   // Stat card bileşeni
   export function StatCard({ label, value, change, icon }: { label: string; value: string | number; change?: string; icon?: React.ReactNode }) {
     const isPositive = change && !change.startsWith('-');
     return (
       <Card className="p-4">
         <div className="flex items-center justify-between">
           <div>
             <p className="text-xs text-slate-400">{label}</p>
             <p className="text-2xl font-bold text-white mt-1">{value}</p>
             {change && (
               <p className={`text-xs mt-1 ${isPositive ? 'text-green-400' : 'text-red-400'}`}>
                 {change} vs last month
               </p>
             )}
           </div>
           {icon && <div className="text-slate-500">{icon}</div>}
         </div>
       </Card>
     );
   }
