=== FIX 2: DATALASTIC.TS KONTROL ===

server/datalastic.ts'de fetch çağrılarının doğru yapıldığını kontrol et:

export async function getVesselPosition(params: { imo?: string; mmsi?: string }): Promise<any | null> {
  const apiKey = process.env.DATALASTIC_API_KEY;
  if (!apiKey) return null;
  
  try {
    const queryParam = params.imo ? `imo=${params.imo}` : `mmsi=${params.mmsi}`;
    const url = `https://api.datalastic.com/api/v0/vessel?api-key=${apiKey}&${queryParam}`;
    
    const response = await fetch(url);
    const data = await response.json();
    
    if (data.meta?.success && data.data) {
      return data.data;
    }
    console.log('Datalastic response not successful:', data);
    return null;
  } catch (error) {
    console.error('Datalastic error:', error);
    return null;
  }
}

export function isDatalasticConfigured(): boolean {
  return !!process.env.DATALASTIC_API_KEY;
}
