Gemi bazlı kapsamlı PMS (Planned Maintenance System) oluştur. Armatör ve Master arasında çalışacak. ISM Code, SOLAS ve Class Society gereksinimlerine uyumlu. Mevcut maintenance, spare-parts ve defect-tracker altyapısını GENİŞLET — silme.

## SİSTEM MİMARİSİ

PMS şu modüllerden oluşacak:
1. Equipment Registry — Gemi ekipman ağacı
2. Maintenance Jobs — Bakım görevleri (zamanlı + saatli)  
3. Work Orders — İş emirleri
4. Running Hours — Çalışma saatleri takibi
5. Spare Parts — Yedek parça yönetimi (mevcut genişletilecek)
6. Defect/Non-Conformity — Arıza raporlama (mevcut genişletilecek)
7. Dry Dock Planning — Havuz planlama (mevcut genişletilecek)
8. Class & Survey — Klas sörveyleri takibi
9. Reports & Analytics — PMS raporları
10. Role-based Access — Armatör vs Master yetkileri

router.get("/vessels/:vesselId/equipment-tree", isAuthenticated, async (req: any, res) => {
  try {
    const vesselId = parseInt(req.params.vesselId);
    const categories = await db.select().from(equipmentCategories)
      .where(eq(equipmentCategories.vesselId, vesselId))
      .orderBy(asc(equipmentCategories.sortOrder));
    const items = await db.select().from(equipmentItems)
      .where(eq(equipmentItems.vesselId, vesselId))
      .orderBy(asc(equipmentItems.sortOrder));
    
    // Ağaç yapısına dönüştür
    function buildTree(parentId: number | null): any[] {
      const cats = categories.filter(c => c.parentId === parentId);
      return cats.map(cat => ({
        ...cat,
        type: "category",
        children: buildTree(cat.id),
        equipment: items.filter(i => i.categoryId === cat.id).map(item => ({
          ...item,
          type: "equipment",
        })),
      }));
    }
    
    res.json({
      tree: buildTree(null),
      flatCategories: categories,
      flatEquipment: items,
      stats: {
        totalCategories: categories.length,
        totalEquipment: items.length,
        withRunningHours: items.filter(i => i.hasRunningHours).length,
        classRelevant: items.filter(i => i.classRelevant).length,
      },
    });
  } catch (e: any) { res.status(500).json({ error: e.message }); }
});

router.get("/vessels/:vesselId/jobs", isAuthenticated, async (req: any, res) => {
  try {
    const vesselId = parseInt(req.params.vesselId);
    const status = req.query.status as string | undefined;
    const priority = req.query.priority as string | undefined;
    const overdue = req.query.overdue === "true";
    
    let query = db.select({
      job: pmsJobs,
      equipmentName: equipmentItems.name,
      equipmentCode: equipmentItems.code,
      categoryName: equipmentCategories.name,
    })
      .from(pmsJobs)
      .leftJoin(equipmentItems, eq(pmsJobs.equipmentId, equipmentItems.id))
      .leftJoin(equipmentCategories, eq(equipmentItems.categoryId, equipmentCategories.id))
      .where(eq(pmsJobs.vesselId, vesselId))
      .orderBy(asc(pmsJobs.nextDueDate))
      .$dynamic();
    
    const rows = await query;
    
    let filtered = rows;
    if (status) filtered = filtered.filter(r => r.job.status === status);
    if (priority) filtered = filtered.filter(r => r.job.priority === priority);
    if (overdue) filtered = filtered.filter(r => r.job.isOverdue);
    
    // Overdue kontrolü güncelle
    const now = new Date();
    for (const row of filtered) {
      const job = row.job;
      let isOd = false;
      let odDays = 0;
      if (job.nextDueDate && new Date(job.nextDueDate) < now) {
        isOd = true;
        odDays = Math.floor((now.getTime() - new Date(job.nextDueDate).getTime()) / 86400000);
      }
      if (job.nextDueRunningHours && job.equipmentId) {
        const equip = await db.select({ rh: equipmentItems.currentRunningHours }).from(equipmentItems).where(eq(equipmentItems.id, job.equipmentId)).limit(1);
        if (equip[0] && equip[0].rh && job.nextDueRunningHours <= equip[0].rh) {
          isOd = true;
        }
      }
      if (isOd !== job.isOverdue || odDays !== job.overdueDays) {
        await db.update(pmsJobs).set({ isOverdue: isOd, overdueDays: odDays }).where(eq(pmsJobs.id, job.id));
        job.isOverdue = isOd;
        job.overdueDays = odDays;
      }
    }
    
    res.json({
      jobs: filtered.map(r => ({ ...r.job, equipmentName: r.equipmentName, equipmentCode: r.equipmentCode, categoryName: r.categoryName })),
      summary: {
        total: filtered.length,
        overdue: filtered.filter(r => r.job.isOverdue).length,
        dueSoon: filtered.filter(r => !r.job.isOverdue && r.job.nextDueDate && new Date(r.job.nextDueDate) <= new Date(now.getTime() + 30 * 86400000)).length,
        active: filtered.filter(r => r.job.status === "active").length,
        completed: filtered.filter(r => r.job.status === "completed").length,
      },
    });
  } catch (e: any) { res.status(500).json({ error: e.message }); }
});

router.patch("/work-orders/:id", isAuthenticated, async (req: any, res) => {
  try {
    const id = parseInt(req.params.id);
    const data = { ...req.body, updatedAt: new Date() };
    
    // Tamamlandıysa PMS job'ı da güncelle
    if (data.status === "completed") {
      data.actualEndDate = data.actualEndDate || new Date();
      data.completedBy = data.completedBy || req.user?.id;
      
      const [wo] = await db.select().from(workOrders).where(eq(workOrders.id, id));
      if (wo?.pmsJobId) {
        const now = new Date();
        let nextDue: Date | null = null;
        const [job] = await db.select().from(pmsJobs).where(eq(pmsJobs.id, wo.pmsJobId));
        if (job) {
          if (job.calendarIntervalDays) {
            nextDue = new Date(now.getTime() + job.calendarIntervalDays * 86400000);
          } else if (job.calendarIntervalMonths) {
            nextDue = new Date(now);
            nextDue.setMonth(nextDue.getMonth() + job.calendarIntervalMonths);
          }
          
          let nextDueRh: number | null = null;
          if (job.runningHoursInterval && wo.runningHoursAtComplete) {
            nextDueRh = wo.runningHoursAtComplete + job.runningHoursInterval;
          }
          
          await db.update(pmsJobs).set({
            lastDoneDate: now,
            lastDoneRunningHours: wo.runningHoursAtComplete,
            lastDoneBy: data.completedBy || req.user?.id,
            lastDoneWorkOrderId: id,
            nextDueDate: nextDue,
            nextDueRunningHours: nextDueRh,
            isOverdue: false,
            overdueDays: 0,
            updatedAt: now,
          }).where(eq(pmsJobs.id, wo.pmsJobId));
        }
      }
    }
    
    const [row] = await db.update(workOrders).set(data).where(eq(workOrders.id, id)).returning();
    res.json(row);
  } catch (e: any) { res.status(500).json({ error: e.message }); }
});

    const [equip] = await db.select().from(equipmentItems).where(eq(equipmentItems.id, equipmentId));
    if (!equip) return res.status(404).json({ message: "Equipment not found" });
    
    const prevRh = equip.currentRunningHours || 0;

    await db.insert(runningHoursLog).values({
      vesselId: equip.vesselId,
      equipmentId,
      recordDate: new Date(),
      runningHours: rh,
      previousRunningHours: prevRh,
      hoursIncrement: rh - prevRh,
      recordedBy: req.user?.id,
      source: "manual",
      notes,
    });
    
    // Equipment güncelle
    await db.update(equipmentItems).set({
      currentRunningHours: rh,
      lastRunningHoursUpdate: new Date(),
      updatedAt: new Date(),
    }).where(eq(equipmentItems.id, equipmentId));
    
    // Bu ekipmana bağlı running hours bazlı PMS job'larını kontrol et
    const jobs = await db.select().from(pmsJobs)
      .where(and(eq(pmsJobs.equipmentId, equipmentId), eq(pmsJobs.intervalType, "running_hours")));
    
    for (const job of jobs) {
      if (job.nextDueRunningHours && rh >= job.nextDueRunningHours) {
        await db.update(pmsJobs).set({ isOverdue: true, updatedAt: new Date() }).where(eq(pmsJobs.id, job.id));
      }
    }
    
    res.json({ equipmentId, previousHours: prevRh, newHours: rh, increment: rh - prevRh });
  } catch (e: any) { res.status(500).json({ error: e.message }); }
});

router.get("/vessels/:vesselId/pms-dashboard", isAuthenticated, async (req: any, res) => {
  try {
    const vesselId = parseInt(req.params.vesselId);
    const now = new Date();
    const thirtyDays = new Date(now.getTime() + 30 * 86400000);
    
    const [jobs, wos, equipment, surveys] = await Promise.all([
      db.select().from(pmsJobs).where(eq(pmsJobs.vesselId, vesselId)),
      db.select().from(workOrders).where(eq(workOrders.vesselId, vesselId)),
      db.select().from(equipmentItems).where(eq(equipmentItems.vesselId, vesselId)),
      db.select().from(classSurveys).where(eq(classSurveys.vesselId, vesselId)),
    ]);
    
    const overdueJobs = jobs.filter(j => j.isOverdue);
    const dueSoonJobs = jobs.filter(j => !j.isOverdue && j.nextDueDate && new Date(j.nextDueDate) <= thirtyDays);
    const openWos = wos.filter(w => w.status === "open" || w.status === "in_progress");
    const upcomingSurveys = surveys.filter(s => s.dueDate && new Date(s.dueDate) <= thirtyDays && s.status !== "completed");
    
    // Aylık tamamlama oranı
    const thisMonth = new Date(now.getFullYear(), now.getMonth(), 1);
    const completedThisMonth = wos.filter(w => w.status === "completed" && w.actualEndDate && new Date(w.actualEndDate) >= thisMonth).length;
    const dueThisMonth = jobs.filter(j => j.nextDueDate && new Date(j.nextDueDate) >= thisMonth && new Date(j.nextDueDate) <= new Date(now.getFullYear(), now.getMonth() + 1, 0)).length;
    
    res.json({
      kpis: {
        totalJobs: jobs.length,
        overdueCount: overdueJobs.length,
        dueSoonCount: dueSoonJobs.length,
        openWorkOrders: openWos.length,
        completionRate: dueThisMonth > 0 ? Math.round((completedThisMonth / dueThisMonth) * 100) : 100,
        totalEquipment: equipment.length,
        upcomingSurveys: upcomingSurveys.length,
      },
      overdueJobs: overdueJobs.slice(0, 10),
      dueSoonJobs: dueSoonJobs.slice(0, 10),
      openWorkOrders: openWos.slice(0, 10),
      upcomingSurveys: upcomingSurveys.slice(0, 5),
      recentlyCompleted: wos.filter(w => w.status === "completed").sort((a, b) => new Date(b.actualEndDate || 0).getTime() - new Date(a.actualEndDate || 0).getTime()).slice(0, 10),
    });
  } catch (e: any) { res.status(500).json({ error: e.message }); }
});
