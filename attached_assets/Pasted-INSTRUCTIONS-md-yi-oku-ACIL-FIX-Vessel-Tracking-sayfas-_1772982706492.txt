=== FIX 3: LEAFLET MARKER İKONLARI ===

Leaflet default marker ikonları genelde bozuk olur. Custom gemi ikonu kullan:

// Gemi marker ikonu
const vesselIcon = L.divIcon({
  className: 'vessel-marker',
  html: `<div style="
    width: 24px; height: 24px; 
    background: #38BDF8; 
    border-radius: 50%; 
    border: 2px solid white;
    display: flex; align-items: center; justify-content: center;
    font-size: 12px; color: white;
    box-shadow: 0 2px 8px rgba(0,0,0,0.3);
  ">🚢</div>`,
  iconSize: [24, 24],
  iconAnchor: [12, 12],
});

// Rotasyona göre dönen gemi ikonu (heading/course kullanarak)
function createVesselIcon(heading: number, speed: number) {
  const color = speed > 0.5 ? '#38BDF8' : '#F59E0B'; // hareket eden mavi, duran sarı
  return L.divIcon({
    className: 'vessel-marker',
    html: `<div style="
      width: 28px; height: 28px;
      transform: rotate(${heading || 0}deg);
      display: flex; align-items: center; justify-content: center;
    ">
      <svg width="28" height="28" viewBox="0 0 24 24" fill="${color}">
        <path d="M12 2L4 20h16L12 2z" opacity="0.9"/>
      </svg>
    </div>`,
    iconSize: [28, 28],
    iconAnchor: [14, 14],
  });
}

=== FIX 4: GEMİLERİ HARİTAYA EKLE ===

Gemiler haritada görünsün:

// Pozisyonları çek
const { data: vessels } = useQuery({
  queryKey: ['/api/vessel-track/fleet'],
  refetchInterval: 60000, // her 1 dakikada güncelle
});

// Marker'ları haritaya ekle
useEffect(() => {
  if (!mapInstanceRef.current || !vessels?.length) return;
  
  const map = mapInstanceRef.current;
  
  // Eski marker'ları temizle
  markersRef.current.forEach(m => map.removeLayer(m));
  markersRef.current = [];
  
  vessels.forEach(vessel => {
    if (!vessel.latitude || !vessel.longitude) return;
    
    const icon = createVesselIcon(vessel.heading || vessel.course || 0, vessel.speed || 0);
    
    const marker = L.marker([vessel.latitude, vessel.longitude], { icon })
      .addTo(map)
      .bindPopup(`
        <div style="min-width: 200px; font-family: sans-serif;">
          <strong>${vessel.name || 'Unknown'}</strong><br/>
          <small>IMO: ${vessel.imo || 'N/A'} | MMSI: ${vessel.mmsi || 'N/A'}</small><br/>
          <hr style="margin: 4px 0"/>
          Speed: ${vessel.speed || 0} kn<br/>
          Course: ${vessel.course || 0}°<br/>
          Destination: ${vessel.destination || 'N/A'}<br/>
          ETA: ${vessel.eta || 'N/A'}<br/>
          Status: ${vessel.navStatus || 'N/A'}<br/>
          <small>Updated: ${vessel.lastUpdate || 'N/A'}</small>
        </div>
      `);
    
    markersRef.current.push(marker);
  });
  
  // Tüm gemileri görecek şekilde zoom
  if (vessels.length > 0) {
    const bounds = L.latLngBounds(vessels.filter(v => v.latitude && v.longitude).map(v => [v.latitude, v.longitude]));
    map.fitBounds(bounds, { padding: [50, 50] });
  }
}, [vessels]);
