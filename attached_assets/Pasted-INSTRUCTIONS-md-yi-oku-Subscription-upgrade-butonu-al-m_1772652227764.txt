İyzico payment gateway entegrre et:

1. Paketi kur: npm install iyzipay

2. Yeni dosya oluştur: server/payment.ts

   import Iyzipay from 'iyzipay';

   const iyzipay = new Iyzipay({
     apiKey: process.env.IYZICO_API_KEY || '',
     secretKey: process.env.IYZICO_SECRET_KEY || '',
     uri: process.env.IYZICO_BASE_URL || 'https://sandbox-api.iyzipay.com',
   });

   interface PaymentRequest {
     userId: string;
     email: string;
     firstName: string;
     lastName: string;
     plan: 'standard' | 'unlimited';
     ip: string;
   }

   const PLAN_PRICES: Record<string, { price: string; name: string }> = {
     standard: { price: '49.90', name: 'VesselPDA Standard Plan' },
     unlimited: { price: '199.90', name: 'VesselPDA Unlimited Plan' },
   };

   export function createCheckoutForm(req: PaymentRequest): Promise<any> {
     return new Promise((resolve, reject) => {
       const plan = PLAN_PRICES[req.plan];
       if (!plan) return reject(new Error('Invalid plan'));

       const request = {
         locale: Iyzipay.LOCALE.EN,
         conversationId: `sub_${req.userId}_${Date.now()}`,
         price: plan.price,
         paidPrice: plan.price,
         currency: Iyzipay.CURRENCY.TRY,
         basketId: `basket_${req.userId}`,
         paymentGroup: Iyzipay.PAYMENT_GROUP.SUBSCRIPTION,
         callbackUrl: `${process.env.APP_URL || 'https://vesselpda.com'}/api/payment/callback`,
         enabledInstallments: [1],
         buyer: {
           id: req.userId,
           name: req.firstName || 'User',
           surname: req.lastName || 'VesselPDA',
           email: req.email,
           identityNumber: '11111111111',
           registrationAddress: 'Istanbul, Turkey',
           ip: req.ip,
           city: 'Istanbul',
           country: 'Turkey',
         },
         billingAddress: {
           contactName: `${req.firstName} ${req.lastName}`,
           city: 'Istanbul',
           country: 'Turkey',
           address: 'Istanbul, Turkey',
         },
         basketItems: [
           {
             id: `plan_${req.plan}`,
             name: plan.name,
             category1: 'Subscription',
             itemType: Iyzipay.BASKET_ITEM_TYPE.VIRTUAL,
             price: plan.price,
           },
         ],
       };

       iyzipay.checkoutFormInitialize.create(request, (err: any, result: any) => {
         if (err) return reject(err);
         resolve(result);
       });
     });
   }

   export function retrieveCheckoutResult(token: string): Promise<any> {
     return new Promise((resolve, reject) => {
       iyzipay.checkoutForm.retrieve({ locale: Iyzipay.LOCALE.EN, token }, (err: any, result: any) => {
         if (err) return reject(err);
         resolve(result);
       });
     });
   }

3. server/routes.ts'e ödeme endpoint'leri ekle:

   import { createCheckoutForm, retrieveCheckoutResult } from './payment';

   // Ödeme başlat
   app.post("/api/payment/checkout", isAuthenticated, async (req: any, res: any, next: any) => {
     try {
       const userId = req.user.claims.sub;
       const user = await storage.getUser(userId);
       if (!user) return res.status(404).json({ error: "User not found" });

       const { plan } = req.body;
       if (!['standard', 'unlimited'].includes(plan)) {
         return res.status(400).json({ error: "Invalid plan" });
       }

       const result = await createCheckoutForm({
         userId,
         email: user.email || '',
         firstName: user.firstName || '',
         lastName: user.lastName || '',
         plan,
         ip: req.ip || '127.0.0.1',
       });

       if (result.status === 'success') {
         res.json({
           checkoutFormContent: result.checkoutFormContent,
           token: result.token,
         });
       } else {
         res.status(400).json({ error: result.errorMessage || 'Payment init failed' });
       }
     } catch (error) {
       next(error);
     }
   });

   // İyzico callback
   app.post("/api/payment/callback", async (req: any, res: any, next: any) => {
     try {
       const { token } = req.body;
       if (!token) return res.redirect('/pricing?status=error');

       const result = await retrieveCheckoutResult(token);

       if (result.status === 'success' && result.paymentStatus === 'SUCCESS') {
         // conversationId'den userId ve plan çıkar
         const conversationId = result.conversationId || '';
         const parts = conversationId.split('_');
         const userId = parts[1];
         
         // basketItems'dan plan belirle
         const basketItem = result.basketItems?.[0];
         let plan = 'standard';
         if (basketItem?.id?.includes('unlimited')) plan = 'unlimited';

         // Kullanıcı planını güncelle
         const limits: Record<string, number> = { standard: 10, unlimited: 999999 };
         await storage.updateUser(userId, {
           subscriptionPlan: plan,
           proformaLimit: limits[plan],
         });

         // Audit log
         if (typeof logAction === 'function') {
           await logAction(userId, 'subscription_upgrade', 'user', userId, { plan, paymentId: result.paymentId });
         }

         res.redirect('/pricing?status=success&plan=' + plan);
       } else {
         res.redirect('/pricing?status=failed');
       }
     } catch (error) {
       console.error('Payment callback error:', error);
       res.redirect('/pricing?status=error');
     }
   });

   // Ödeme durumu kontrol
   app.get("/api/payment/status", isAuthenticated, async (req: any, res: any) => {
     const userId = req.user.claims.sub;
     const user = await storage.getUser(userId);
     res.json({
       plan: user?.subscriptionPlan || 'free',
       proformaLimit: user?.proformaLimit || 1,
       proformaCount: user?.proformaCount || 0,
     });
   });

4. Frontend — client/src/pages/pricing.tsx güncelle:

   Mevcut upgrade butonlarını gerçek ödeme flow'una bağla:
   
   const handleUpgrade = async (plan: 'standard' | 'unlimited') => {
     try {
       const res = await fetch('/api/payment/checkout', {
         method: 'POST',
         headers: { 'Content-Type': 'application/json' },
         body: JSON.stringify({ plan }),
       });
       const data = await res.json();
       
       if (data.checkoutFormContent) {
         // İyzico checkout form'u modal'da göster
         const div = document.createElement('div');
         div.innerHTML = data.checkoutFormContent;
         document.body.appendChild(div);
         // İyzico kendi JS'ini yükleyip formu render eder
       } else {
         toast({ title: "Error", description: data.error, variant: "destructive" });
       }
     } catch (error) {
       toast({ title: "Error", description: "Payment service unavailable", variant: "destructive" });
     }
   };

   // URL'den ödeme sonucu kontrol et
   useEffect(() => {
     const params = new URLSearchParams(window.location.search);
     const status = params.get('status');
     if (status === 'success') {
       toast({ title: "Başarılı!", description: "Plan yükseltildi" });
       queryClient.invalidateQueries({ queryKey: ['/api/payment/status'] });
     } else if (status === 'failed') {
       toast({ title: "Başarısız", description: "Ödeme tamamlanamadı", variant: "destructive" });
     }
   }, []);
