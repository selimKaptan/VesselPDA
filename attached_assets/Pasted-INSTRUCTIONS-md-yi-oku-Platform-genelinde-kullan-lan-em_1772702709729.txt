// PDA Gönderimi
export function pdaSentTemplate(params: {
  recipientName: string;
  vesselName: string;
  portName: string;
  referenceNumber: string;
  totalUsd: number;
  totalEur: number;
  senderName: string;
  senderCompany: string;
  approveUrl: string;
  revisionUrl: string;
  message?: string;
}): { subject: string; html: string } {
  const content = `
    <h1>Proforma Disbursement Account</h1>
    <p>Dear ${params.recipientName},</p>
    <p>${params.message || `Please find below the Proforma Disbursement Account for your review.`}</p>
    
    <div class="divider"></div>
    
    <table style="width:100%">
      <tr><td class="info-label">Reference</td><td class="info-value">${params.referenceNumber}</td></tr>
      <tr><td class="info-label">Vessel</td><td class="info-value">${params.vesselName}</td></tr>
      <tr><td class="info-label">Port</td><td class="info-value">${params.portName}</td></tr>
    </table>
    
    <div style="text-align:center; padding: 20px 0;">
      <div class="amount">$${params.totalUsd.toLocaleString()}</div>
      <p style="margin:4px 0">€${params.totalEur.toLocaleString()}</p>
    </div>
    
    <div style="text-align:center">
      <a href="${params.approveUrl}" class="btn btn-green">✓ Approve PDA</a>
      &nbsp;&nbsp;
      <a href="${params.revisionUrl}" class="btn btn-red">Request Revision</a>
    </div>
    
    <div class="divider"></div>
    <p style="font-size:12px; color:#475569">A PDF copy is attached to this email.</p>
    <p style="font-size:12px; color:#475569">Sent by ${params.senderName} — ${params.senderCompany}</p>
  `;
  return {
    subject: `PDA for Review: ${params.vesselName} at ${params.portName} [${params.referenceNumber}]`,
    html: baseTemplate(content),
  };
}

=== 3. MEVCUT EMAIL FONKSİYONLARINI GÜNCELLE ===

server/email.ts dosyasındaki mevcut email fonksiyonlarını HTML template'lere geçir:

Her email gönderme fonksiyonunda:
ESKİ: html: `<p>Plain text email content</p>`
YENİ: İlgili template fonksiyonunu import edip kullan

Örnek:
import { pdaSentTemplate, newTenderTemplate, bidReceivedTemplate, nominationTemplate, welcomeTemplate, invoiceCreatedTemplate, norTenderedTemplate, fdaReadyTemplate } from './email-templates';

// sendProformaEmail fonksiyonunda:
const { subject, html } = pdaSentTemplate({
  recipientName: recipientName,
  vesselName: vessel.name,
  portName: port.name,
  referenceNumber: proforma.referenceNumber,
  totalUsd: proforma.totalUsd,
  totalEur: proforma.totalEur,
  senderName: user.firstName + ' ' + user.lastName,
  senderCompany: companyProfile?.companyName || '',
  approveUrl: `https://vesselpda.com/api/proformas/${proforma.id}/approve?token=${token}`,
  revisionUrl: `https://vesselpda.com/api/proformas/${proforma.id}/request-revision?token=${token}`,
  message: customMessage,
});

await resend.emails.send({
  from: 'VesselPDA <noreply@vesselpda.com>',
  to: recipientEmail,
  subject: subject,
  html: html,
});

Aynı pattern'i şu fonksiyonlara uygula:
- sendNominationEmail → nominationTemplate
- sendBidReceivedEmail → bidReceivedTemplate
- sendNewTenderEmail → newTenderTemplate
- sendContactEmail → basit baseTemplate
- sendMessageBridgeEmail → basit baseTemplate
- sendForumReplyEmail → basit baseTemplate

=== 4. EMAIL PREVIEW (ADMIN) ===

Admin panele email template önizleme ekle:

GET /api/admin/email-preview/:template
- Template adını alır, örnek data ile HTML döner
- Admin tarayıcıda template'in nasıl göründüğünü görebilir

Admin sayfasına "Email Templates" bölümü:
- Template listesi (PDA, Tender, Bid, Nomination, NOR, Invoice, FDA, Welcome)
- Her birinin yanında "Preview" butonu
- Preview tıklanınca yeni sekmede örnek data ile HTML göster

=== 5. IMPORTANT ===

- Email HTML inline CSS kullanmalı (email client'lar external CSS desteklemez)
- Koyu tema: maritime renkler (lacivert arka plan, beyaz text)
- Logo: text-based (⚓ VesselPDA) — resim embed etme
- Mobile responsive: max-width: 600px
- CTA butonları büyük ve belirgin
- Mevcut email gönderme altyapısını (Resend) koru
- from adresi: 'VesselPDA <noreply@vesselpda.com>' (veya mevcut ne ise)
- Mevcut çalışan email'leri BOZMA, sadece HTML'i güzelleştir
- Template fonksiyonları pure function — side effect yok
