useEffect(() => {
  if (!mapContainerRef.current || mapInstanceRef.current) return;
  
  let cancelled = false;
  
  async function initMap() {
    // Önce token'ı al
    const token = await getMapboxToken();
    
    if (cancelled || !mapContainerRef.current) return;
    
    const map = L.map(mapContainerRef.current, {
      center: [39.0, 27.0],

      attributionControl: true,
    });

    if (token && token.length > 10) {
      L.tileLayer(
        `https://api.mapbox.com/styles/v1/mapbox/navigation-night-v1/tiles/256/{z}/{x}/{y}@2x?access_token=${token}`,
        {
          attribution: '© <a href="https://www.mapbox.com/">Mapbox</a> © <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a>',
          maxZoom: 19,
          tileSize: 256,
        }
      ).addTo(map);
      console.log('Map: Using Mapbox premium tiles');
    } else {
      // Fallback: CartoDB dark tiles (ücretsiz)
      L.tileLayer('https://{s}.basemaps.cartocdn.com/dark_all/{z}/{x}/{y}{r}.png', {
        attribution: '© OpenStreetMap © CARTO',
        maxZoom: 19,
        subdomains: 'abcd',
      }).addTo(map);
      console.log('Map: Using CartoDB fallback tiles');
    }
    
    // OpenSeaMap deniz işaretleri overlay
    L.tileLayer('https://tiles.openseamap.org/seamark/{z}/{x}/{y}.png', {
      attribution: '© OpenSeaMap',
      maxZoom: 18,
      opacity: 0.7,
    }).addTo(map);
    
    // Kontroller
    L.control.zoom({ position: 'topright' }).addTo(map);
    L.control.scale({ maxWidth: 100, imperial: false, position: 'bottomleft' }).addTo(map);
    
    mapInstanceRef.current = map;
    
    // Container boyutu geç hesaplanırsa fix
    setTimeout(() => map.invalidateSize(), 300);
    setTimeout(() => map.invalidateSize(), 1000);
  }
  
  initMap();
  
  return () => {
    cancelled = true;
    if (mapInstanceRef.current) {
      mapInstanceRef.current.remove();
      mapInstanceRef.current = null;
    }
  };
}, []);

=== EK: getMapboxToken FONKSİYONUNU DA DÜZELT ===

Dosyanın üstündeki getMapboxToken fonksiyonunu kontrol et — /api/config/mapbox endpoint'i var mı? Yoksa sadece env'den oku:

async function getMapboxToken(): Promise<string> {
  // 1. Vite env'den oku (en hızlı)
  const envToken = (import.meta.env.VITE_MAPBOX_TOKEN as string) || "";
  if (envToken && envToken.length > 10) {
    console.log('Mapbox token: from env');
    return envToken;
  }
  
  // 2. Backend'den çek (fallback)
  try {
    const r = await fetch("/api/config/mapbox", { credentials: "include" });
    if (r.ok) {
      const data = await r.json();
      if (data.token && data.token.length > 10) {
        console.log('Mapbox token: from backend');
        return data.token;
      }
    }
  } catch (e) {
    console.log('Mapbox token: backend fetch failed');
  }
  
  console.log('Mapbox token: not found, will use fallback tiles');
  return "";
}
