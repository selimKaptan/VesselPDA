=== 1. BACKEND — SEARCH ENDPOINT ===

server/routes/ içinde uygun modüle veya yeni search.routes.ts oluştur:

GET /api/search?q=xxx&limit=10

Bu endpoint TÜM kaynaklarda arama yapar ve kategorize sonuç döner:

app.get("/api/search", isAuthenticated, async (req: any, res: any, next: any) => {
  try {
    const query = (req.query.q as string || '').trim();
    if (!query || query.length < 2) return res.json({ results: [] });
    
    const userId = req.user.claims.sub;
    const user = await storage.getUser(userId);
    const limit = parseInt(req.query.limit as string) || 5;
    const searchTerm = `%${query}%`;

    // Parallel arama — tüm kaynaklar aynı anda
    const [vessels, ports, proformas, tenders, voyages, companies, forumTopics] = await Promise.all([
      // Vessels
      db.select({ id: vesselsTable.id, name: vesselsTable.name, flag: vesselsTable.flag, imoNumber: vesselsTable.imoNumber })
        .from(vesselsTable)
        .where(or(
          ilike(vesselsTable.name, searchTerm),
          ilike(vesselsTable.imoNumber, searchTerm),
          ilike(vesselsTable.callSign, searchTerm)
        ))
        .limit(limit),

      // Ports
      db.select({ id: portsTable.id, name: portsTable.name, country: portsTable.country, code: portsTable.code })
        .from(portsTable)
        .where(or(
          ilike(portsTable.name, searchTerm),
          ilike(portsTable.code, searchTerm),
          ilike(portsTable.country, searchTerm)
        ))
        .limit(limit),

      // Proformas
      db.select({ id: proformasTable.id, referenceNumber: proformasTable.referenceNumber, totalUsd: proformasTable.totalUsd, status: proformasTable.status })
        .from(proformasTable)
        .where(and(
          user?.userRole === 'admin' ? undefined : eq(proformasTable.userId, userId),
          ilike(proformasTable.referenceNumber, searchTerm)
        ))
        .limit(limit),

      // Tenders
      db.select({ id: portTendersTable.id, vesselName: portTendersTable.vesselName, status: portTendersTable.status })
        .from(portTendersTable)
        .where(ilike(portTendersTable.vesselName, searchTerm))
        .limit(limit),

      // Voyages
      db.select({ id: voyagesTable.id, status: voyagesTable.status })
        .from(voyagesTable)
        .where(and(
          user?.userRole === 'admin' ? undefined : or(
            eq(voyagesTable.userId, userId),
            eq(voyagesTable.agentUserId, userId)
          )
        ))
        .limit(limit),

      // Companies
      db.select({ id: companyProfilesTable.id, companyName: companyProfilesTable.companyName, companyType: companyProfilesTable.companyType })
        .from(companyProfilesTable)
        .where(and(
          eq(companyProfilesTable.isApproved, true),
          ilike(companyProfilesTable.companyName, searchTerm)
        ))
        .limit(limit),

      // Forum Topics
      db.select({ id: forumTopicsTable.id, title: forumTopicsTable.title })
        .from(forumTopicsTable)
        .where(ilike(forumTopicsTable.title, searchTerm))
        .limit(limit),
    ]);

    // Kategorize et
    const results = [
      ...vessels.map(v => ({ type: 'vessel', id: v.id, title: v.name, subtitle: `${v.flag || ''} — IMO: ${v.imoNumber || 'N/A'}`, icon: '🚢', href: '/vessels' })),
      ...ports.map(p => ({ type: 'port', id: p.id, title: p.name, subtitle: `${p.country || ''} — ${p.code || ''}`, icon: '⚓', href: `/port-info` })),
      ...proformas.map(p => ({ type: 'proforma', id: p.id, title: p.referenceNumber || `PDA #${p.id}`, subtitle: `$${p.totalUsd || 0} — ${p.status}`, icon: '📋', href: `/proformas/${p.id}` })),
      ...tenders.map(t => ({ type: 'tender', id: t.id, title: t.vesselName || `Tender #${t.id}`, subtitle: t.status, icon: '📢', href: `/tenders/${t.id}` })),
      ...companies.map(c => ({ type: 'company', id: c.id, title: c.companyName, subtitle: c.companyType, icon: '🏢', href: `/directory/${c.id}` })),
      ...forumTopics.map(f => ({ type: 'forum', id: f.id, title: f.title, subtitle: 'Forum Topic', icon: '💬', href: `/forum/${f.id}` })),
    ];

    res.json({ results, query });
  } catch (error) { next(error); }
});

=== 4. KEYBOARD SHORTCUT ===

App genelinde keyboard listener ekle:
- Ctrl+K veya Cmd+K → GlobalSearch aç
- Escape → GlobalSearch kapat

useEffect(() => {
  const handler = (e: KeyboardEvent) => {
    if ((e.metaKey || e.ctrlKey) && e.key === 'k') {
      e.preventDefault();
      setSearchOpen(true);
    }
  };
  window.addEventListener('keydown', handler);
  return () => window.removeEventListener('keydown', handler);
}, []);
